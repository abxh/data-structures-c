179 return NULL;

379static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value) {
380 assert(hashtable_ptr != NULL);
381 assert(JOIN(FHASHTABLE_NAME, contains_key)(hashtable_ptr, key) == false);
382 assert(JOIN(FHASHTABLE_NAME, is_full)(hashtable_ptr) == false);
383
384 const size_t index_mask = hashtable_ptr->capacity - 1;
385 const size_t key_hash = HASH_FUNCTION(key);
386
387 size_t index = key_hash & index_mask;
388 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
389
390 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
391
392 // swap if current offset is larger. will ensure the maximum
393 // offset (from the ideal position) is minimized.
394
395 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
396
397 FHASHTABLE_SLOT_TYPE temp = hashtable_ptr->slots[index];
398 hashtable_ptr->slots[index] = current_slot;
399 current_slot = temp;
400 }
401 index++;
402 index &= index_mask;
403 current_slot.offset++;
404 }
405 hashtable_ptr->slots[index] = current_slot;
406 hashtable_ptr->count++;

469static inline bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key) {
470 assert(hashtable_ptr != NULL);
471
472 const size_t index_mask = hashtable_ptr->capacity - 1;
473 const size_t key_hash = HASH_FUNCTION(key);
474
475 size_t index = key_hash & index_mask;
476 size_t max_possible_offset = 0;
477
478 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
479
480 max_possible_offset <= hashtable_ptr->slots[index].offset) {
481
482 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
483
484 // mark as deleted:
485 hashtable_ptr->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
486 hashtable_ptr->count--;
487
488 // reduce offsets as much as possible by moving back offset elements:
489
490 size_t next_index = (index + 1) & index_mask;
491
492 while (hashtable_ptr->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
493
494 hashtable_ptr->slots[next_index].offset > 0) {
495
496 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
497 hashtable_ptr->slots[index].offset--;
498
499 hashtable_ptr->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
500
501 index = next_index;
502 next_index = (index + 1) & index_mask;
503 }
504 return true;
505 }
506 index = (index + 1) & index_mask;
507 max_possible_offset++;
