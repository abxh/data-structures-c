214static inline FHASHTABLE_TYPE* JOIN(FHASHTABLE_NAME, create)(const uint32_t capacity)
215{
216 if (capacity == 0 || capacity > UINT32_MAX / 4) {
217 return NULL;
218 }
219 const uint32_t capacity_new = round_up_pow2_32(capacity);
220 FHASHTABLE_TYPE* hashtable_ptr =
221 (FHASHTABLE_TYPE*)malloc(offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));
222 if (!hashtable_ptr) {
223 return NULL;
224 }
225 memset(hashtable_ptr, 0, offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));
226
227 hashtable_ptr->count = 0;
228 hashtable_ptr->capacity = capacity_new;
229
230 for (uint32_t i = 0; i < hashtable_ptr->capacity; i++) {
231 hashtable_ptr->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
232 }
233
234 return hashtable_ptr;
235}

290static inline bool JOIN(FHASHTABLE_NAME, contains_key)(const FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key)
291{
292 assert(hashtable_ptr != NULL);
293
294 const uint32_t key_hash = HASH_FUNCTION(key);
295 const uint32_t index_mask = hashtable_ptr->capacity - 1;
296
297 uint32_t index = key_hash & index_mask;
298 uint32_t max_possible_offset = 0;
299
300 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
301
302 max_possible_offset <= hashtable_ptr->slots[index].offset) {
303
304 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
305 return true;
306 }
307
308 index++;
309 index &= index_mask;
310 max_possible_offset++;
311 }
312 return false;
313}

420static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value)
421{
422 assert(hashtable_ptr != NULL);
423 assert(FHASHTABLE_CONTAINS_KEY(hashtable_ptr, key) == false);
424
425 const uint32_t index_mask = hashtable_ptr->capacity - 1;
426 const uint32_t key_hash = HASH_FUNCTION(key);
427
428 uint32_t index = key_hash & index_mask;
429 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
430
431 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
432
433 // swap if current offset is larger. will ensure the maximum
434 // offset (from the ideal position) is minimized.
435
436 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
437
438 FHASHTABLE_SLOT_TYPE temp = hashtable_ptr->slots[index];
439 hashtable_ptr->slots[index] = current_slot;
440 current_slot = temp;
441 }
442 index++;
443 index &= index_mask;
444 current_slot.offset++;
445 }
446 hashtable_ptr->slots[index] = current_slot;
447 hashtable_ptr->count++;
448}

461static inline void JOIN(FHASHTABLE_NAME, update)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value)
462{
463 assert(hashtable_ptr != NULL);
464
465 const uint32_t index_mask = hashtable_ptr->capacity - 1;
466 const uint32_t key_hash = HASH_FUNCTION(key);
467
468 uint32_t index = key_hash & index_mask;

509static inline bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key)
510{
511 assert(hashtable_ptr != NULL);
512
513 const uint32_t index_mask = hashtable_ptr->capacity - 1;
514 const uint32_t key_hash = HASH_FUNCTION(key);
515
516 uint32_t index = key_hash & index_mask;
517 uint32_t max_possible_offset = 0;
518
519 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
520
521 max_possible_offset <= hashtable_ptr->slots[index].offset) {
522
523 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
524
525 // mark as deleted:
526 hashtable_ptr->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
527 hashtable_ptr->count--;
528
529 // reduce offsets as much as possible by moving back offset elements:
530
531 uint32_t next_index = (index + 1) & index_mask;
532
533 while (hashtable_ptr->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
534
535 hashtable_ptr->slots[next_index].offset > 0) {
536
537 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
538 hashtable_ptr->slots[index].offset--;
539
540 hashtable_ptr->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
541
542 index = next_index;
543 next_index = (index + 1) & index_mask;
544 }
545 return true;
546 }
547 index = (index + 1) & index_mask;
548 max_possible_offset++;
549 }
550 return false;
551}
