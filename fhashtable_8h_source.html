214static inline FHASHTABLE_TYPE* JOIN(FHASHTABLE_NAME, create)(const size_t capacity) {
215 if (capacity == 0 || capacity > UINT32_MAX / 4) {
216 return NULL;
217 }
218 const size_t capacity_new = round_up_pow2(capacity);
219 if (capacity_new > (SIZE_MAX - offsetof(FHASHTABLE_TYPE, slots)) / sizeof(FHASHTABLE_SLOT_TYPE)) {
220 return NULL;
221 }
222
223 FHASHTABLE_TYPE* hashtable_ptr =
224 (FHASHTABLE_TYPE*)malloc(offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));
225 if (!hashtable_ptr) {
226 return NULL;
227 }
228 memset(hashtable_ptr, 0, offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));
229
230 hashtable_ptr->count = 0;
231 hashtable_ptr->capacity = capacity_new;
232
233 for (size_t i = 0; i < hashtable_ptr->capacity; i++) {
234 hashtable_ptr->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
235 }
236
237 return hashtable_ptr;
238}

290static inline bool JOIN(FHASHTABLE_NAME, contains_key)(const FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key) {
291 assert(hashtable_ptr != NULL);
292
293 const size_t key_hash = HASH_FUNCTION(key);
294 const size_t index_mask = hashtable_ptr->capacity - 1;
295
296 size_t index = key_hash & index_mask;
297 size_t max_possible_offset = 0;
298
299 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
300
301 max_possible_offset <= hashtable_ptr->slots[index].offset) {
302
303 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
304 return true;
305 }
306
307 index++;
308 index &= index_mask;
309 max_possible_offset++;
310 }
311 return false;
312}

416static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value) {
417 assert(hashtable_ptr != NULL);
418 assert(JOIN(FHASHTABLE_NAME, contains_key)(hashtable_ptr, key) == false);
419
420 const size_t index_mask = hashtable_ptr->capacity - 1;
421 const size_t key_hash = HASH_FUNCTION(key);
422
423 size_t index = key_hash & index_mask;
424 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
425
426 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
427
428 // swap if current offset is larger. will ensure the maximum
429 // offset (from the ideal position) is minimized.
430
431 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
432
433 FHASHTABLE_SLOT_TYPE temp = hashtable_ptr->slots[index];
434 hashtable_ptr->slots[index] = current_slot;
435 current_slot = temp;
436 }
437 index++;
438 index &= index_mask;
439 current_slot.offset++;
440 }
441 hashtable_ptr->slots[index] = current_slot;
442 hashtable_ptr->count++;
443}

503static inline bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key) {
504 assert(hashtable_ptr != NULL);
505
506 const size_t index_mask = hashtable_ptr->capacity - 1;
507 const size_t key_hash = HASH_FUNCTION(key);
508
509 size_t index = key_hash & index_mask;
510 size_t max_possible_offset = 0;
511
512 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
513
514 max_possible_offset <= hashtable_ptr->slots[index].offset) {
515
516 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
517
518 // mark as deleted:
519 hashtable_ptr->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
520 hashtable_ptr->count--;
521
522 // reduce offsets as much as possible by moving back offset elements:
523
524 size_t next_index = (index + 1) & index_mask;
525
526 while (hashtable_ptr->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
527
528 hashtable_ptr->slots[next_index].offset > 0) {
529
530 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
531 hashtable_ptr->slots[index].offset--;
532
533 hashtable_ptr->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
534
535 index = next_index;
536 next_index = (index + 1) & index_mask;
537 }
538 return true;
539 }
540 index = (index + 1) & index_mask;
541 max_possible_offset++;
542 }
543 return false;
544}
