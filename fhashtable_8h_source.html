185static inline FHASHTABLE_TYPE* JOIN(FHASHTABLE_NAME, create)(const size_t capacity) {
186 const size_t capacity_new0 = capacity * FHASHTABLE_CAPACITY_FACTOR;
187 if (capacity_new0 < capacity || capacity_new0 > UINTPTR_MAX / 4) {
188 return NULL;
189 }
190 const size_t capacity_new1 = round_up_pow2(capacity);
191 if (capacity_new1 > (SIZE_MAX - offsetof(FHASHTABLE_TYPE, slots)) / sizeof(FHASHTABLE_SLOT_TYPE)) {
192 return NULL;
193 }
194
195 FHASHTABLE_TYPE* hashtable_ptr = malloc(offsetof(FHASHTABLE_TYPE, slots) + capacity_new1 * sizeof(FHASHTABLE_SLOT_TYPE));
196 if (!hashtable_ptr) {
197 return NULL;
198 }
199 memset(hashtable_ptr, 0, offsetof(FHASHTABLE_TYPE, slots) + capacity_new1 * sizeof(FHASHTABLE_SLOT_TYPE));
200
201 hashtable_ptr->count = 0;
202 hashtable_ptr->capacity = capacity_new1;
203
204 for (size_t i = 0; i < hashtable_ptr->capacity; i++) {
205 hashtable_ptr->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;

388static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value) {
389 assert(hashtable_ptr != NULL);
390 assert(JOIN(FHASHTABLE_NAME, contains_key)(hashtable_ptr, key) == false);
391 assert(JOIN(FHASHTABLE_NAME, is_full)(hashtable_ptr) == false);
392
393 const size_t index_mask = hashtable_ptr->capacity - 1;
394 const size_t key_hash = HASH_FUNCTION(key);
395
396 size_t index = key_hash & index_mask;
397 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
398
399 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
400
401 // swap if current offset is larger. will ensure the maximum
402 // offset (from the ideal position) is minimized.
403
404 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
405
406 FHASHTABLE_SLOT_TYPE temp = hashtable_ptr->slots[index];
407 hashtable_ptr->slots[index] = current_slot;
408 current_slot = temp;
409 }
410 index++;
411 index &= index_mask;
412 current_slot.offset++;
413 }
414 hashtable_ptr->slots[index] = current_slot;
415 hashtable_ptr->count++;

478static inline bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key) {
479 assert(hashtable_ptr != NULL);
480
481 const size_t index_mask = hashtable_ptr->capacity - 1;
482 const size_t key_hash = HASH_FUNCTION(key);
483
484 size_t index = key_hash & index_mask;
485 size_t max_possible_offset = 0;
486
487 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
488
489 max_possible_offset <= hashtable_ptr->slots[index].offset) {
490
491 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
492
493 // mark as deleted:
494 hashtable_ptr->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
495 hashtable_ptr->count--;
496
497 // reduce offsets as much as possible by moving back offset elements:
498
499 size_t next_index = (index + 1) & index_mask;
500
501 while (hashtable_ptr->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
502
503 hashtable_ptr->slots[next_index].offset > 0) {
504
505 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
506 hashtable_ptr->slots[index].offset--;
507
508 hashtable_ptr->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
509
510 index = next_index;
511 next_index = (index + 1) & index_mask;
512 }
513 return true;
514 }
515 index = (index + 1) & index_mask;
516 max_possible_offset++;
517 }
