194static inline FHASHTABLE_TYPE* JOIN(FHASHTABLE_NAME, create)(const size_t capacity) {
195 if (capacity < 2 || capacity > UINT32_MAX / 4) {
196 return NULL;
197 }
198 const size_t capacity_new = round_up_pow2(capacity);
199 if (capacity_new > (SIZE_MAX - offsetof(FHASHTABLE_TYPE, slots)) / sizeof(FHASHTABLE_SLOT_TYPE)) {
200 return NULL;
201 }
202
203 FHASHTABLE_TYPE* hashtable_ptr =
204 (FHASHTABLE_TYPE*)malloc(offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));

206 return NULL;

400static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value) {
401 assert(hashtable_ptr != NULL);
402 assert(JOIN(FHASHTABLE_NAME, contains_key)(hashtable_ptr, key) == false);
403
404 const size_t index_mask = hashtable_ptr->capacity - 1;
405 const size_t key_hash = HASH_FUNCTION(key);
406
407 size_t index = key_hash & index_mask;
408 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
409
410 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
411
412 // swap if current offset is larger. will ensure the maximum
413 // offset (from the ideal position) is minimized.

415 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
416

418 hashtable_ptr->slots[index] = current_slot;
419 current_slot = temp;

423 current_slot.offset++;
