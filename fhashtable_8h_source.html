199static inline FHASHTABLE_TYPE* JOIN(FHASHTABLE_NAME, create)(const uint32_t capacity)
200{
201 if (capacity == 0 || capacity > UINT32_MAX / 4) {
202 return NULL;
203 }
204 const uint32_t capacity_new = round_up_pow2_32(capacity);
205
206 FHASHTABLE_TYPE* hashtable_ptr =
207 (FHASHTABLE_TYPE*)calloc(1, offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));
208
209 if (!hashtable_ptr) {
210 return NULL;
211 }
212
213 hashtable_ptr->count = 0;
214 hashtable_ptr->capacity = capacity_new;
215
216 for (uint32_t i = 0; i < hashtable_ptr->capacity; i++) {
217 hashtable_ptr->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
218 }
219
220 return hashtable_ptr;
221}

280static inline bool JOIN(FHASHTABLE_NAME, contains_key)(const FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key)
281{
282 assert(hashtable_ptr != NULL);
283
284 const uint32_t key_hash = HASH_FUNCTION(key);
285 const uint32_t index_mask = hashtable_ptr->capacity - 1;
286
287 uint32_t index = key_hash & index_mask;
288 uint32_t max_possible_offset = 0;
289
290 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
291
292 max_possible_offset <= hashtable_ptr->slots[index].offset) {
293
294 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
295 return true;
296 }
297
298 index++;
299 index &= index_mask;
300 max_possible_offset++;
301 }
302 return false;

412static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value)
413{
414 assert(hashtable_ptr != NULL);
415 assert(FHASHTABLE_CONTAINS_KEY(hashtable_ptr, key) == false);
416
417 const uint32_t index_mask = hashtable_ptr->capacity - 1;
418 const uint32_t key_hash = HASH_FUNCTION(key);
419
420 uint32_t index = key_hash & index_mask;
421 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
422
423 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
424
425 // swap if current offset is larger. will ensure the maximum
426 // offset (from the ideal position) is minimized.
427
428 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
429
430 FHASHTABLE_SLOT_TYPE temp = hashtable_ptr->slots[index];
431 hashtable_ptr->slots[index] = current_slot;
432 current_slot = temp;
433 }
434 index++;
435 index &= index_mask;
436 current_slot.offset++;
437 }
438 hashtable_ptr->slots[index] = current_slot;
439 hashtable_ptr->count++;
440}

453static inline void JOIN(FHASHTABLE_NAME, update)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value)
454{
455 assert(hashtable_ptr != NULL);
456
457 const uint32_t index_mask = hashtable_ptr->capacity - 1;
458 const uint32_t key_hash = HASH_FUNCTION(key);
459
460 uint32_t index = key_hash & index_mask;

502static inline bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key)
503{
504 assert(hashtable_ptr != NULL);
505
506 const uint32_t index_mask = hashtable_ptr->capacity - 1;
507 const uint32_t key_hash = HASH_FUNCTION(key);
508
509 uint32_t index = key_hash & index_mask;
510 uint32_t max_possible_offset = 0;
511
512 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
513
514 max_possible_offset <= hashtable_ptr->slots[index].offset) {
515
516 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
517
518 // mark as deleted:
519 hashtable_ptr->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
520 hashtable_ptr->count--;
521
522 // reduce offsets as much as possible by moving back offset elements:
523
524 uint32_t next_index = (index + 1) & index_mask;
525
526 while (hashtable_ptr->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
527
528 hashtable_ptr->slots[next_index].offset > 0) {
529
530 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
531 hashtable_ptr->slots[index].offset--;
532
533 hashtable_ptr->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
534
535 index = next_index;
536 next_index = (index + 1) & index_mask;
537 }
538 return true;
539 }
540 index = (index + 1) & index_mask;
541 max_possible_offset++;
542 }
543 return false;
544}
