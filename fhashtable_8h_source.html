198static inline FHASHTABLE_TYPE* JOIN(FHASHTABLE_NAME, create)(const uint32_t capacity)
199{
200 if (capacity == 0 || capacity > UINT32_MAX / 4) {
201 return NULL;
202 }
203 const uint32_t capacity_new = round_up_pow2_32(capacity);
204
205 FHASHTABLE_TYPE* hashtable_ptr =
206 (FHASHTABLE_TYPE*)calloc(1, offsetof(FHASHTABLE_TYPE, slots) + capacity_new * sizeof(FHASHTABLE_SLOT_TYPE));
207
208 if (!hashtable_ptr) {
209 return NULL;
210 }
211
212 hashtable_ptr->count = 0;
213 hashtable_ptr->capacity = capacity_new;
214
215 for (uint32_t i = 0; i < hashtable_ptr->capacity; i++) {
216 hashtable_ptr->slots[i].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
217 }
218
219 return hashtable_ptr;
220}

267static inline bool JOIN(FHASHTABLE_NAME, contains_key)(const FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key)
268{
269 assert(hashtable_ptr != NULL);
270
271 const uint32_t key_hash = HASH_FUNCTION(key);
272 const uint32_t index_mask = hashtable_ptr->capacity - 1;
273
274 uint32_t index = key_hash & index_mask;
275 uint32_t max_possible_offset = 0;
276
277 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
278
279 max_possible_offset <= hashtable_ptr->slots[index].offset) {
280
281 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
282 return true;
283 }
284
285 index++;
286 index &= index_mask;
287 max_possible_offset++;
288 }
289 return false;

386static inline void JOIN(FHASHTABLE_NAME, insert)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value)
387{
388 assert(hashtable_ptr != NULL);
389 assert(FHASHTABLE_CONTAINS_KEY(hashtable_ptr, key) == false);
390
391 const uint32_t index_mask = hashtable_ptr->capacity - 1;
392 const uint32_t key_hash = HASH_FUNCTION(key);
393
394 uint32_t index = key_hash & index_mask;
395 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
396
397 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {
398
399 // swap if current offset is larger. will ensure the maximum
400 // offset (from the ideal position) is minimized.
401
402 if (current_slot.offset > hashtable_ptr->slots[index].offset) {
403
404 FHASHTABLE_SLOT_TYPE temp = hashtable_ptr->slots[index];
405 hashtable_ptr->slots[index] = current_slot;
406 current_slot = temp;
407 }
408 index++;
409 index &= index_mask;
410 current_slot.offset++;
411 }
412 hashtable_ptr->slots[index] = current_slot;
413 hashtable_ptr->count++;
414}

423static inline void JOIN(FHASHTABLE_NAME, update)(FHASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value)
424{
425 assert(hashtable_ptr != NULL);
426
427 const uint32_t index_mask = hashtable_ptr->capacity - 1;
428 const uint32_t key_hash = HASH_FUNCTION(key);
429
430 uint32_t index = key_hash & index_mask;
431 FHASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};

433 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET) {

469static inline bool JOIN(FHASHTABLE_NAME, delete)(FHASHTABLE_TYPE* hashtable_ptr, const KEY_TYPE key)
470{
471 assert(hashtable_ptr != NULL);
472
473 const uint32_t index_mask = hashtable_ptr->capacity - 1;
474 const uint32_t key_hash = HASH_FUNCTION(key);
475
476 uint32_t index = key_hash & index_mask;
477 uint32_t max_possible_offset = 0;
478
479 while (hashtable_ptr->slots[index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
480
481 max_possible_offset <= hashtable_ptr->slots[index].offset) {
482
483 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
484
485 // mark as deleted:
486 hashtable_ptr->slots[index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
487 hashtable_ptr->count--;
488
489 // reduce offsets as much as possible by moving back offset elements:
490
491 uint32_t next_index = (index + 1) & index_mask;
492
493 while (hashtable_ptr->slots[next_index].offset != FHASHTABLE_EMPTY_SLOT_OFFSET &&
494
495 hashtable_ptr->slots[next_index].offset > 0) {
496
497 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
498 hashtable_ptr->slots[index].offset--;
499
500 hashtable_ptr->slots[next_index].offset = FHASHTABLE_EMPTY_SLOT_OFFSET;
501
502 index = next_index;
503 next_index = (index + 1) & index_mask;
504 }
505 return true;
506 }
507 index = (index + 1) & index_mask;
508 max_possible_offset++;
509 }
510 return false;
511}
