164static inline FPQUEUE_TYPE* JOIN(FPQUEUE_NAME, create)(const uint32_t capacity) {
165 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FPQUEUE_TYPE, elements)) / sizeof(FPQUEUE_ELEMENT_TYPE)) {
166 return NULL;
167 }
168 FPQUEUE_TYPE* pqueue_ptr = (FPQUEUE_TYPE*)malloc(offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));
169
170 if (!pqueue_ptr) {
171 return NULL;
172 }
173 memset(pqueue_ptr, 0, offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));
174
175 pqueue_ptr->count = 0;
176 pqueue_ptr->capacity = capacity;
177
178 return pqueue_ptr;
179}

253static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE* pqueue_ptr, const uint32_t index) {
254 assert(pqueue_ptr != NULL);
255 assert(pqueue_ptr->count == 0 || index < pqueue_ptr->count);
256
257 const uint32_t l = fpqueue_left_child(index);
258 const uint32_t r = fpqueue_right_child(index);
259
260 uint32_t largest = index;
261 if (l < pqueue_ptr->count && pqueue_ptr->elements[l].priority > pqueue_ptr->elements[index].priority) {
262 largest = l;
263 }
264 if (r < pqueue_ptr->count && pqueue_ptr->elements[r].priority > pqueue_ptr->elements[largest].priority) {
265 largest = r;
266 }
267
268 if (largest == index) {
269 return;
270 }
271
272 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
273 pqueue_ptr->elements[index] = pqueue_ptr->elements[largest];
274 pqueue_ptr->elements[largest] = temp;
275
276 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, largest);
277}

290static inline VALUE_TYPE JOIN(FPQUEUE_NAME, pop_max)(FPQUEUE_TYPE* pqueue_ptr) {
291 assert(pqueue_ptr != NULL);

306static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE* pqueue_ptr, uint32_t index) {
307 assert(pqueue_ptr != NULL);
308 assert(index < pqueue_ptr->count);
309
310 uint32_t parent;
311 while (index > 0 && (parent = fpqueue_parent(index),
312
313 pqueue_ptr->elements[parent].priority < pqueue_ptr->elements[index].priority)) {
314
315 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
316 pqueue_ptr->elements[index] = pqueue_ptr->elements[parent];
317 pqueue_ptr->elements[parent] = temp;
318
319 index = parent;
320 }
321}

335static inline void JOIN(FPQUEUE_NAME, push)(FPQUEUE_TYPE* pqueue_ptr, const VALUE_TYPE value, const uint32_t priority) {
336 assert(pqueue_ptr != NULL);
337 assert(FPQUEUE_IS_FULL(pqueue_ptr) == false);
338
339 const uint32_t index = pqueue_ptr->count;

374 assert(src_pqueue_ptr != NULL);
