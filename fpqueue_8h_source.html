164static inline FPQUEUE_TYPE* JOIN(FPQUEUE_NAME, create)(const uint32_t capacity)
165{
166 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FPQUEUE_TYPE, elements)) / sizeof(FPQUEUE_ELEMENT_TYPE)) {
167 return NULL;
168 }
169 FPQUEUE_TYPE* pqueue_ptr = (FPQUEUE_TYPE*)malloc(offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));
170
171 if (!pqueue_ptr) {
172 return NULL;
173 }
174 memset(pqueue_ptr, 0, offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));
175
176 pqueue_ptr->count = 0;
177 pqueue_ptr->capacity = capacity;
178
179 return pqueue_ptr;
180}

259static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE* pqueue_ptr, const uint32_t index)
260{
261 assert(pqueue_ptr != NULL);
262 assert(pqueue_ptr->count == 0 || index < pqueue_ptr->count);
263
264 const uint32_t l = fpqueue_left_child(index);
265 const uint32_t r = fpqueue_right_child(index);
266
267 uint32_t largest = index;
268 if (l < pqueue_ptr->count && pqueue_ptr->elements[l].priority > pqueue_ptr->elements[index].priority) {
269 largest = l;
270 }
271 if (r < pqueue_ptr->count && pqueue_ptr->elements[r].priority > pqueue_ptr->elements[largest].priority) {
272 largest = r;
273 }
274
275 if (largest == index) {
276 return;
277 }
278
279 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
280 pqueue_ptr->elements[index] = pqueue_ptr->elements[largest];
281 pqueue_ptr->elements[largest] = temp;
282
283 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, largest);
284}

297static inline VALUE_TYPE JOIN(FPQUEUE_NAME, pop_max)(FPQUEUE_TYPE* pqueue_ptr)
298{
299 assert(pqueue_ptr != NULL);

303
304 pqueue_ptr->elements[0] = pqueue_ptr->elements[pqueue_ptr->count - 1];
305
306 pqueue_ptr->count--;

308 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, 0);

314static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE* pqueue_ptr, uint32_t index)
315{
316 assert(pqueue_ptr != NULL);
317 assert(index < pqueue_ptr->count);
318
319 uint32_t parent;
320 while (index > 0 && (parent = fpqueue_parent(index),
321
322 pqueue_ptr->elements[parent].priority < pqueue_ptr->elements[index].priority)) {
323
324 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
325 pqueue_ptr->elements[index] = pqueue_ptr->elements[parent];
326 pqueue_ptr->elements[parent] = temp;
327
328 index = parent;
329 }
330}

344static inline void JOIN(FPQUEUE_NAME, push)(FPQUEUE_TYPE* pqueue_ptr, VALUE_TYPE value, uint32_t priority)

384static inline void JOIN(FPQUEUE_NAME, copy)(FPQUEUE_TYPE* restrict dest_pqueue_ptr, const FPQUEUE_TYPE* restrict src_pqueue_ptr)
385{
