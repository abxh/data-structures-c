157static inline FPQUEUE_TYPE* JOIN(FPQUEUE_NAME, create)(const uint32_t capacity)
158{
159 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FPQUEUE_TYPE, elements)) / sizeof(FPQUEUE_ELEMENT_TYPE)) {
160 return NULL;
161 }
162 FPQUEUE_TYPE* pqueue_ptr = (FPQUEUE_TYPE*)malloc(offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));
163
164 if (!pqueue_ptr) {
165 return NULL;
166 }
167 memset(pqueue_ptr, 0, offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));
168
169 pqueue_ptr->count = 0;
170 pqueue_ptr->capacity = capacity;
171
172 return pqueue_ptr;
173}

257static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE* pqueue_ptr, const uint32_t index)
258{
259 assert(pqueue_ptr != NULL);
260 assert(pqueue_ptr->count == 0 || index < pqueue_ptr->count);
261
262 const uint32_t l = fpqueue_left_child(index);
263 const uint32_t r = fpqueue_right_child(index);
264
265 uint32_t largest = index;
266 if (l < pqueue_ptr->count && pqueue_ptr->elements[l].priority > pqueue_ptr->elements[index].priority) {
267 largest = l;
268 }
269 if (r < pqueue_ptr->count && pqueue_ptr->elements[r].priority > pqueue_ptr->elements[largest].priority) {
270 largest = r;
271 }
272
273 if (largest == index) {
274 return;
275 }
276
277 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
278 pqueue_ptr->elements[index] = pqueue_ptr->elements[largest];
279 pqueue_ptr->elements[largest] = temp;
280
281 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, largest);
282}

296static inline VALUE_TYPE JOIN(FPQUEUE_NAME, pop_max)(FPQUEUE_TYPE* pqueue_ptr)
297{
298 assert(pqueue_ptr != NULL);

313static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE* pqueue_ptr, uint32_t index)
314{
315 assert(pqueue_ptr != NULL);
316 assert(index < pqueue_ptr->count);
317
318 uint32_t parent;
319 while (index > 0 && (parent = fpqueue_parent(index),
320
321 pqueue_ptr->elements[parent].priority < pqueue_ptr->elements[index].priority)) {
322
323 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
324 pqueue_ptr->elements[index] = pqueue_ptr->elements[parent];
325 pqueue_ptr->elements[parent] = temp;
326
327 index = parent;
328 }
329}

343static inline void JOIN(FPQUEUE_NAME, push)(FPQUEUE_TYPE* pqueue_ptr, VALUE_TYPE value, uint32_t priority)
344{

383static inline void JOIN(FPQUEUE_NAME, copy)(FPQUEUE_TYPE* restrict dest_pqueue_ptr, const FPQUEUE_TYPE* restrict src_pqueue_ptr)
384{
