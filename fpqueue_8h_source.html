155static inline FPQUEUE_TYPE* JOIN(FPQUEUE_NAME, create)(const uint32_t capacity)
156{
157 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FPQUEUE_TYPE, elements)) / sizeof(FPQUEUE_ELEMENT_TYPE)) {
158 return NULL;
159 }
160 FPQUEUE_TYPE* pqueue_ptr = (FPQUEUE_TYPE*)calloc(1, offsetof(FPQUEUE_TYPE, elements) + capacity * sizeof(FPQUEUE_ELEMENT_TYPE));

163 return NULL;
164 }

238static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE* pqueue_ptr, const uint32_t index)
239{
240 assert(pqueue_ptr != NULL);
241 assert(pqueue_ptr->count == 0 || index < pqueue_ptr->count);
242
243 const uint32_t l = fpqueue_left_child(index);
244 const uint32_t r = fpqueue_right_child(index);
245
246 uint32_t largest = index;
247 if (l < pqueue_ptr->count && pqueue_ptr->elements[l].priority > pqueue_ptr->elements[index].priority) {
248 largest = l;
249 }
250 if (r < pqueue_ptr->count && pqueue_ptr->elements[r].priority > pqueue_ptr->elements[largest].priority) {
251 largest = r;
252 }
253
254 if (largest == index) {
255 return;
256 }
257
258 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
259 pqueue_ptr->elements[index] = pqueue_ptr->elements[largest];
260 pqueue_ptr->elements[largest] = temp;
261
262 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, largest);
263}

272static inline VALUE_TYPE JOIN(FPQUEUE_NAME, pop_max)(FPQUEUE_TYPE* pqueue_ptr)
273{
274 assert(pqueue_ptr != NULL);
275 assert(FPQUEUE_IS_EMPTY(pqueue_ptr) == false);

277 VALUE_TYPE max = pqueue_ptr->elements[0].value;
278
279 pqueue_ptr->elements[0] = pqueue_ptr->elements[pqueue_ptr->count - 1];
280
281 pqueue_ptr->count--;
282
283 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, 0);

289static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE* pqueue_ptr, uint32_t index)
290{
291 assert(pqueue_ptr != NULL);
292 assert(index < pqueue_ptr->count);
293
294 uint32_t parent;
295 while (index > 0 && (parent = fpqueue_parent(index),
296
297 pqueue_ptr->elements[parent].priority < pqueue_ptr->elements[index].priority)) {
298
299 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
300 pqueue_ptr->elements[index] = pqueue_ptr->elements[parent];
301 pqueue_ptr->elements[parent] = temp;
302
303 index = parent;
304 }
305}

315static inline void JOIN(FPQUEUE_NAME, push)(FPQUEUE_TYPE* pqueue_ptr, VALUE_TYPE value, uint32_t priority)
316{
317 assert(pqueue_ptr != NULL);
318 assert(FPQUEUE_IS_FULL(pqueue_ptr) == false);
319
320 const uint32_t index = pqueue_ptr->count;
321
322 pqueue_ptr->elements[index] = (FPQUEUE_ELEMENT_TYPE){.priority = priority, .value = value};
323
324 pqueue_ptr->count++;
325
326 JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(pqueue_ptr, index);

347static inline void JOIN(FPQUEUE_NAME, copy)(FPQUEUE_TYPE* restrict dest_pqueue_ptr, const FPQUEUE_TYPE* restrict src_pqueue_ptr)
348{
349 assert(src_pqueue_ptr != NULL);
350 assert(dest_pqueue_ptr != NULL);
351 assert(src_pqueue_ptr->count <= dest_pqueue_ptr->capacity);
352 assert(FPQUEUE_IS_EMPTY(dest_pqueue_ptr));

355 dest_pqueue_ptr->elements[i] = src_pqueue_ptr->elements[i];

357 dest_pqueue_ptr->count = src_pqueue_ptr->count;
