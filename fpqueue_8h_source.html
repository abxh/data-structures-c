367static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE *self, uint32_t index)
368{
369 assert(self != NULL);
370 assert(index < self->count);
371
372 uint32_t parent;
373 while (index > 0) {
374 parent = fpqueue_parent(index);
375
376 const bool sorted = self->elements[parent].priority >= self->elements[index].priority;
377
378 if (sorted) {
379 break;
380 }
381
382 const FPQUEUE_ELEMENT_TYPE temp = self->elements[index];
383 self->elements[index] = self->elements[parent];
384 self->elements[parent] = temp;
385
386 index = parent;
387 }
388}

390static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE *self, const uint32_t index)
391{
392 assert(self != NULL);
393 assert(self->count == 0 || index < self->count);
394
395 const uint32_t l = fpqueue_left_child(index);
396 const uint32_t r = fpqueue_right_child(index);
397
398 uint32_t largest = index;
399 if (l < self->count && self->elements[l].priority > self->elements[index].priority) {
400 largest = l;
401 }
402 if (r < self->count && self->elements[r].priority > self->elements[largest].priority) {
403 largest = r;
404 }
405
406 if (largest == index) {
407 return;
408 }
409
410 const FPQUEUE_ELEMENT_TYPE temp = self->elements[index];
411 self->elements[index] = self->elements[largest];
412 self->elements[largest] = temp;
413
414 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(self, largest);
415}
