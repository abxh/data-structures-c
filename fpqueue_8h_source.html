308static inline void JOIN(FPQUEUE_NAME, copy)(FPQUEUE_TYPE* restrict dest_pqueue_ptr, const FPQUEUE_TYPE* restrict src_pqueue_ptr)
309{
310 assert(src_pqueue_ptr != NULL);

323static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE* pqueue_ptr, uint32_t index)
324{
325 assert(pqueue_ptr != NULL);
326 assert(index < pqueue_ptr->count);
327
328 uint32_t parent;
329 while (index > 0 && (parent = fpqueue_parent(index),
330
331 pqueue_ptr->elements[parent].priority < pqueue_ptr->elements[index].priority)) {
332
333 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
334 pqueue_ptr->elements[index] = pqueue_ptr->elements[parent];
335 pqueue_ptr->elements[parent] = temp;
336
337 index = parent;
338 }
339}

341static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE* pqueue_ptr, const uint32_t index)
342{
343 assert(pqueue_ptr != NULL);
344 assert(pqueue_ptr->count == 0 || index < pqueue_ptr->count);
345
346 const uint32_t l = fpqueue_left_child(index);
347 const uint32_t r = fpqueue_right_child(index);
348
349 uint32_t largest = index;
350 if (l < pqueue_ptr->count && pqueue_ptr->elements[l].priority > pqueue_ptr->elements[index].priority) {
351 largest = l;
352 }
353 if (r < pqueue_ptr->count && pqueue_ptr->elements[r].priority > pqueue_ptr->elements[largest].priority) {
354 largest = r;
355 }
356
357 if (largest == index) {
358 return;
359 }
360
361 const FPQUEUE_ELEMENT_TYPE temp = pqueue_ptr->elements[index];
362 pqueue_ptr->elements[index] = pqueue_ptr->elements[largest];
363 pqueue_ptr->elements[largest] = temp;
364
365 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(pqueue_ptr, largest);
366}
