325static inline bool JOIN(__QUEUE_PREFIX, enqueue)(QUEUE_TYPE* queue_ptr, VALUE_TYPE value) {
326 assert(NULL != queue_ptr);
327
328 size_t old_capacity = queue_ptr->index_mask + 1;
329 if (queue_ptr->count == old_capacity) {
330 if (old_capacity > SIZE_MAX / sizeof(VALUE_TYPE) / 2) {
331 return false;
332 }
333 size_t new_capacity = 2 * old_capacity;
334 void* temp_ptr =
335 queue_ptr->allocator_ops.reallocate_f_p(queue_ptr->allocator_context_ptr, queue_ptr->values, alignof(VALUE_TYPE),
336 sizeof(VALUE_TYPE) * old_capacity, sizeof(VALUE_TYPE) * new_capacity);
337 if (temp_ptr == NULL) {
338 return false;
339 }
340 queue_ptr->values = (VALUE_TYPE*)temp_ptr;
341 queue_ptr->index_mask = new_capacity - 1;
342 if (queue_ptr->end_index <= queue_ptr->start_index) {
343 size_t n = queue_ptr->end_index;
344 memcpy(&queue_ptr->values[old_capacity], &queue_ptr->values[0], sizeof(VALUE_TYPE) * n);
345 queue_ptr->end_index += old_capacity;
346 }
347 }
348 queue_ptr->values[queue_ptr->end_index] = value;
349 queue_ptr->end_index = (queue_ptr->end_index + 1) & queue_ptr->index_mask;
350 queue_ptr->count++;
351
352 return true;
353}
