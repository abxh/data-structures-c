179 if (!queue_ptr) {

182 size_t capacity = queue_ptr->index_mask + 1;
183 QUEUE_TYPE* other_queue_ptr =
184 JOIN(__QUEUE_PREFIX, create_with_specified)(capacity, queue_ptr->allocator_context_ptr, queue_ptr->allocator_ops);
185 if (!other_queue_ptr) {
186 return NULL;

345static inline bool JOIN(__QUEUE_PREFIX, enqueue)(QUEUE_TYPE* queue_ptr, VALUE_TYPE value) {
346 assert(NULL != queue_ptr);
347
348 size_t old_capacity = queue_ptr->index_mask + 1;
349 if (queue_ptr->count + 1 == old_capacity) {
350 if (old_capacity > SIZE_MAX / sizeof(VALUE_TYPE) / 2) {
351 return false;
352 }
353 size_t new_capacity = 2 * old_capacity;
354 void* temp_ptr =
355 queue_ptr->allocator_ops.reallocate_f_p(queue_ptr->allocator_context_ptr, queue_ptr->values, alignof(VALUE_TYPE),
356 sizeof(VALUE_TYPE) * old_capacity, sizeof(VALUE_TYPE) * new_capacity);
357 if (temp_ptr == NULL) {
358 return false;
359 }
360 queue_ptr->values = (VALUE_TYPE*)temp_ptr;
361 queue_ptr->index_mask = new_capacity - 1;
362 if (queue_ptr->end_index < queue_ptr->start_index) {
363 size_t n = queue_ptr->end_index;
364 size_t m = (old_capacity - queue_ptr->start_index);
365
366 assert(n + m == queue_ptr->count);
367
368 if (n <= m) {
369 memcpy(&queue_ptr->values[old_capacity], &queue_ptr->values[0], n);
370 queue_ptr->end_index += old_capacity;
371 } else {
372 memcpy(&queue_ptr->values[new_capacity - m], &queue_ptr->values[queue_ptr->start_index], m);
373 queue_ptr->start_index += old_capacity;
374 }
375 }
376 }
377 queue_ptr->values[queue_ptr->end_index] = value;
378 queue_ptr->end_index = (queue_ptr->end_index + 1) & queue_ptr->index_mask;
379 queue_ptr->count++;
380
381 return true;
382}
