344static inline bool JOIN(__QUEUE_PREFIX, enqueue)(QUEUE_TYPE* queue_ptr, VALUE_TYPE value) {
345 assert(NULL != queue_ptr);
346
347 size_t old_capacity = queue_ptr->index_mask + 1;
348 if (queue_ptr->count + 1 == old_capacity) {
349 if (old_capacity > SIZE_MAX / sizeof(VALUE_TYPE) / 2) {
350 return false;
351 }
352 size_t new_capacity = 2 * old_capacity;
353 void* temp_ptr =
354 queue_ptr->allocator_ops.reallocate_f_p(queue_ptr->allocator_context_ptr, queue_ptr->values, alignof(VALUE_TYPE),
355 sizeof(VALUE_TYPE) * old_capacity, sizeof(VALUE_TYPE) * new_capacity);
356 if (temp_ptr == NULL) {
357 return false;
358 }
359 queue_ptr->values = (VALUE_TYPE*)temp_ptr;
360 queue_ptr->index_mask = new_capacity - 1;
361 if (queue_ptr->end_index < queue_ptr->start_index) {
362 size_t n = queue_ptr->end_index;
363 size_t m = (old_capacity - queue_ptr->start_index);
364
365 assert(n + m == queue_ptr->count);
366
367 if (n <= m) {
368 memcpy(&queue_ptr->values[old_capacity], &queue_ptr->values[0], n);
369 queue_ptr->end_index += old_capacity;
370 } else {
371 memcpy(&queue_ptr->values[new_capacity - m], &queue_ptr->values[queue_ptr->start_index], m);
372 queue_ptr->start_index += old_capacity;
373 }
374 }
375 }
376 queue_ptr->values[queue_ptr->end_index] = value;
377 queue_ptr->end_index = (queue_ptr->end_index + 1) & queue_ptr->index_mask;
378 queue_ptr->count++;
379
380 return true;
381}
