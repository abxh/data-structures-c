179static inline QUEUE_TYPE* JOIN(QUEUE_PREFIX, clone)(const QUEUE_TYPE* queue_ptr) {
180 if (!queue_ptr) {
181 return NULL;
182 }
183 size_t capacity = queue_ptr->index_mask + 1;
184 QUEUE_TYPE* other_queue_ptr =
185 JOIN(QUEUE_PREFIX, create_with_specified)(capacity, queue_ptr->allocator_context_ptr, queue_ptr->allocator_ops);
186 if (!other_queue_ptr) {
187 return NULL;

348static inline VALUE_TYPE JOIN(QUEUE_PREFIX, grow)(QUEUE_TYPE* queue_ptr, size_t new_capacity) {
349 assert(NULL != queue_ptr);
350
351 size_t old_capacity = queue_ptr->index_mask + 1;
352 if (!is_pow2(new_capacity) || new_capacity <= old_capacity || new_capacity > SIZE_MAX / sizeof(VALUE_TYPE)) {
353 return false;
354 }
355 void* temp_ptr =
356 queue_ptr->allocator_ops.reallocate_f_ptr(queue_ptr->allocator_context_ptr, queue_ptr->values, alignof(VALUE_TYPE),
357 sizeof(VALUE_TYPE) * old_capacity, sizeof(VALUE_TYPE) * new_capacity);
358 if (temp_ptr == NULL) {
359 return false;
360 }
361 queue_ptr->values = (VALUE_TYPE*)temp_ptr;
362 queue_ptr->index_mask = new_capacity - 1;
363 if (queue_ptr->end_index < queue_ptr->start_index) {
364 size_t n = queue_ptr->end_index;
365 size_t m = (old_capacity - queue_ptr->start_index);
366
367 assert(n + m == queue_ptr->count);
368
369 if (n <= m) {
370 memcpy(&queue_ptr->values[old_capacity], &queue_ptr->values[0], n);
371 queue_ptr->end_index += old_capacity;
372 } else {
373 memcpy(&queue_ptr->values[new_capacity - m], &queue_ptr->values[queue_ptr->start_index], m);
374 queue_ptr->start_index = new_capacity - m;
375 }
376 }
377 return true;
378}

393static inline bool JOIN(QUEUE_PREFIX, enqueue)(QUEUE_TYPE* queue_ptr, VALUE_TYPE value) {
394 assert(NULL != queue_ptr);
395
396 size_t old_capacity = queue_ptr->index_mask + 1;
397 if (queue_ptr->count + 1 == old_capacity && !JOIN(QUEUE_PREFIX, grow)(queue_ptr, 2 * old_capacity)) {
398 return false;
399 }
400 queue_ptr->values[queue_ptr->end_index] = value;
