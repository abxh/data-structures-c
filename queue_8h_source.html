121 allocator_ops_type allocator_ops) {

123 return NULL;
124 }
125 QUEUE_TYPE* queue_ptr = (QUEUE_TYPE*)allocator_ops.allocate_f_ptr(allocator_context_ptr, alignof(QUEUE_TYPE), sizeof(QUEUE_TYPE));
126 if (!queue_ptr) {

132 allocator_ops.deallocate_f_ptr(allocator_context_ptr, queue_ptr);
133 return NULL;
134 }

136 queue_ptr->index_mask = initial_pow2_capacity - 1;
137 queue_ptr->start_index = 0;
138 queue_ptr->end_index = 0;
139

141 queue_ptr->allocator_ops = allocator_ops;

181static inline QUEUE_TYPE* JOIN(QUEUE_PREFIX, clone)(const QUEUE_TYPE* queue_ptr) {
182 if (!queue_ptr) {
183 return NULL;
184 }
185 size_t capacity = queue_ptr->index_mask + 1;
186 QUEUE_TYPE* other_queue_ptr =
187 JOIN(QUEUE_PREFIX, create_with_specified)(capacity, queue_ptr->allocator_context_ptr, queue_ptr->allocator_ops);
188 if (!other_queue_ptr) {
189 return NULL;
190 }

202 }

204 other_queue_ptr->start_index = queue_ptr->start_index;

350static inline VALUE_TYPE JOIN(QUEUE_PREFIX, grow)(QUEUE_TYPE* queue_ptr, size_t new_capacity) {
351 assert(NULL != queue_ptr);
352
353 size_t old_capacity = queue_ptr->index_mask + 1;
354 if (!is_pow2(new_capacity) || new_capacity <= old_capacity || new_capacity > SIZE_MAX / sizeof(VALUE_TYPE)) {
355 return false;
356 }
357 void* temp_ptr =
358 queue_ptr->allocator_ops.reallocate_f_ptr(queue_ptr->allocator_context_ptr, queue_ptr->values, alignof(VALUE_TYPE),
359 sizeof(VALUE_TYPE) * old_capacity, sizeof(VALUE_TYPE) * new_capacity);
360 if (temp_ptr == NULL) {
361 return false;
362 }
363 queue_ptr->values = (VALUE_TYPE*)temp_ptr;
364 queue_ptr->index_mask = new_capacity - 1;
365 if (queue_ptr->end_index < queue_ptr->start_index) {
366 size_t n = queue_ptr->end_index;
367 size_t m = (old_capacity - queue_ptr->start_index);
368
369 assert(n + m == queue_ptr->count);
370
371 if (n <= m) {
372 memcpy(&queue_ptr->values[old_capacity], &queue_ptr->values[0], sizeof(VALUE_TYPE) * n);
373 queue_ptr->end_index += old_capacity;
374 } else {
375 memcpy(&queue_ptr->values[new_capacity - m], &queue_ptr->values[queue_ptr->start_index], sizeof(VALUE_TYPE) * m);
376 queue_ptr->start_index = new_capacity - m;
377 }
378 }
379 return true;
380}

395static inline bool JOIN(QUEUE_PREFIX, enqueue)(QUEUE_TYPE* queue_ptr, VALUE_TYPE value) {
396 assert(NULL != queue_ptr);
397
398 size_t old_capacity = queue_ptr->index_mask + 1;
399 if (queue_ptr->count + 1 == old_capacity && !JOIN(QUEUE_PREFIX, grow)(queue_ptr, 2 * old_capacity)) {
400 return false;
401 }
402 queue_ptr->values[queue_ptr->end_index] = value;
403 queue_ptr->end_index = (queue_ptr->end_index + 1) & queue_ptr->index_mask;
404 queue_ptr->count++;
