94static inline void *arena_allocate_aligned(struct arena *self, const size_t alignment, const size_t size)
95{
96 assert(self);
97
98 void *ptr = (void *)&self->buf_ptr[self->curr_offset];
99
100 size_t space_left = self->buf_len - (size_t)self->curr_offset;
101
102 if (!align(alignment, size, &ptr, &space_left)) {
103 return NULL;
104 }
105
106 const uintptr_t relative_offset = (unsigned char *)ptr - &self->buf_ptr[0];
107
108 self->prev_offset = relative_offset;
109 self->curr_offset = relative_offset + size;
110
111 memset(ptr, 0, size);
112
113 return ptr;
114}

133static inline void *internal_arena_optimize_w_prev_offset(struct arena *self, unsigned char *old_ptr,
134 const size_t old_size, const size_t new_size)
135{
136 if (&self->buf_ptr[self->prev_offset] != old_ptr) {
137 return NULL;
138 }
139
140 self->curr_offset = self->prev_offset + new_size;
141
142 if (new_size > old_size) {
143 const size_t diff = new_size - old_size;
144
145 memset(&self->buf_ptr[self->curr_offset], 0, diff);
146 }
147
148 return old_ptr;
149}

165static inline void *arena_reallocate_aligned(struct arena *self, void *old_ptr_, const size_t alignment,
166 const size_t old_size, const size_t new_size)
167{
168 assert(self);
169 assert(is_pow2(alignment));
170
171 unsigned char *old_ptr = (unsigned char *)old_ptr_;
172
173 if (old_ptr == NULL || old_size == 0) {
174 return arena_allocate_aligned(self, alignment, new_size);
175 }
176
177 const bool inside_buf = self->buf_ptr <= old_ptr && old_ptr < self->buf_ptr + self->buf_len;
178
179 if (!inside_buf) {
180 return NULL;
181 }
182
183 if (internal_arena_optimize_w_prev_offset(self, old_ptr, old_size, new_size)) {
184 return old_ptr;
185 }
186
187 const size_t copy_size = old_size < new_size ? old_size : new_size;
188
189 void *new_mem = arena_allocate_aligned(self, alignment, new_size);
190
191 memmove(new_mem, old_ptr, copy_size);
192
193 return new_mem;
194}
