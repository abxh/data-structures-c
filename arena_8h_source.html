99static inline void* arena_allocate_aligned(arena_type* arena_ptr, const size_t alignment, const size_t size)
100{
101 // clang-format off
102
103 // Align `current_offset` forward to the specified alignment
104
105 const uintptr_t current_ptr =
106 + (uintptr_t) arena_ptr->buffer_ptr
107 + (uintptr_t) arena_ptr->current_offset;
108
109 const uintptr_t relative_offset =
110 + (uintptr_t) align_forward(current_ptr, alignment)
111 - (uintptr_t) arena_ptr->buffer_ptr;
112
113 // clang-format on
114
115 // Check to see if the backing memory has space left
116 if (relative_offset + size <= arena_ptr->buffer_length) {
117
118 arena_ptr->previous_offset = relative_offset;
119 arena_ptr->current_offset = relative_offset + size;
120
121 void* ptr = (unsigned char*)&arena_ptr->buffer_ptr[relative_offset];
122 memset(ptr, 0, size); // zero memory by default
123
124 return ptr;
125 }
126 return NULL;
127}

155static inline void* arena_reallocate_aligned(arena_type* arena_ptr, void* old_memory_ptr, const size_t alignment,
156 const size_t old_size, const size_t new_size)
157{
158 assert(is_pow2(alignment));
159
160 const unsigned char* old_mem_buf = (unsigned char*)old_memory_ptr;
161
162 if (old_mem_buf == NULL || old_size == 0) {
163 return arena_allocate_aligned(arena_ptr, alignment, new_size);
164 }
165 else if (!(arena_ptr->buffer_ptr <= old_mem_buf && old_mem_buf < arena_ptr->buffer_ptr + arena_ptr->buffer_length)) {
166 return NULL;
167 }
168
169 // optimization using previous_offset
170 if (arena_ptr->buffer_ptr + arena_ptr->previous_offset == old_mem_buf) {
171 arena_ptr->current_offset = arena_ptr->previous_offset + new_size;
172 if (new_size > old_size) {
173 memset(&arena_ptr->buffer_ptr[arena_ptr->current_offset], 0, new_size - old_size);
174 }
175 return old_memory_ptr;
176 }
177
178 const size_t copy_size = old_size < new_size ? old_size : new_size;
179
180 void* new_memory = arena_allocate_aligned(arena_ptr, alignment, new_size);
181
182 // Copy across old memory to the new memory
183 memmove(new_memory, old_mem_buf, copy_size);
184
185 return new_memory;
186}
