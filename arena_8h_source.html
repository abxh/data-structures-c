88/* align pointer to the next alignment boundary */
89static inline uintptr_t internal_align_forward(const uintptr_t ptr, const size_t align)
90{
91 assert(is_pow2(align));
92
93 const uintptr_t p = ptr;
94 const uintptr_t a = (uintptr_t)align;
95 const uintptr_t r = p & (a - 1); // Same as (p % a) but faster as 'a' is a power of two
96
97 // % is defined so for r = (p % a), q: some integer
98 // p = q * a + r
99
100 // assuming r != 0:
101 //
102 // (q+1)*a
103 // ----------->
104 // r a-r
105 // ----> ----->
106 // q*a p p+a-r
107 // <------|----|------|------>
108
109 if (r != 0) {
110 // If 'p' address is not aligned, push the address to the
111 // next value which is aligned
112 return p + a - r;
113 }
114 else {
115 return p;
116 }
117}

129static inline void* arena_allocate_aligned(arena_type* arena_ptr, const size_t alignment, const size_t size)
130{
131 // clang-format off
132
133 // Align `current_offset` forward to the specified alignment
134
135 const uintptr_t current_ptr =
136 + (uintptr_t) arena_ptr->buffer_ptr
137 + (uintptr_t) arena_ptr->current_offset;
138
139 const uintptr_t relative_offset =
140 + (uintptr_t) internal_align_forward(current_ptr, alignment)
141 - (uintptr_t) arena_ptr->buffer_ptr;
142
143 // clang-format on
144
145 // Check to see if the backing memory has space left
146 if (relative_offset + size <= arena_ptr->buffer_length) {
147
148 arena_ptr->previous_offset = relative_offset;
149 arena_ptr->current_offset = relative_offset + size;
150
151 void* ptr = (unsigned char*)&arena_ptr->buffer_ptr[relative_offset];
152 memset(ptr, 0, size); // zero memory by default
153
154 return ptr;
155 }
156 return NULL;
157}

185static inline void* arena_reallocate_aligned(arena_type* arena_ptr, void* old_memory_ptr, const size_t alignment,
186 const size_t old_size, const size_t new_size)
187{
188 assert(is_pow2(alignment));
189
190 const unsigned char* old_mem_buf = (unsigned char*)old_memory_ptr;
191
192 if (old_mem_buf == NULL || old_size == 0) {
193 return arena_allocate_aligned(arena_ptr, alignment, new_size);
194 }
195 else if (!(arena_ptr->buffer_ptr <= old_mem_buf && old_mem_buf < arena_ptr->buffer_ptr + arena_ptr->buffer_length)) {
196 return NULL;
197 }
198
199 // optimization using previous_offset
200 if (arena_ptr->buffer_ptr + arena_ptr->previous_offset == old_mem_buf) {
201 arena_ptr->current_offset = arena_ptr->previous_offset + new_size;
202 if (new_size > old_size) {
203 memset(&arena_ptr->buffer_ptr[arena_ptr->current_offset], 0, new_size - old_size);
204 }
205 return old_memory_ptr;
206 }
207
208 const size_t copy_size = old_size < new_size ? old_size : new_size;
209
210 void* new_memory = arena_allocate_aligned(arena_ptr, alignment, new_size);
211
212 // Copy across old memory to the new memory
213 memmove(new_memory, old_mem_buf, copy_size);
214
215 return new_memory;
216}
