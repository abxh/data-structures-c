127 assert(is_pow2(alignment));
128
129 const unsigned char* old_mem_buf = (unsigned char*)old_memory_ptr;
130
131 if (old_mem_buf == NULL || old_size == 0) {
132 return arena_allocate_aligned(arena_ptr, alignment, new_size);
133 } else if (!(arena_ptr->buffer_ptr <= old_mem_buf && old_mem_buf < arena_ptr->buffer_ptr + arena_ptr->buffer_length)) {
134 return NULL;
135 }
136
137 // optimization using previous_offset
138 if (arena_ptr->buffer_ptr + arena_ptr->previous_offset == old_mem_buf) {
139 arena_ptr->current_offset = arena_ptr->previous_offset + new_size;
140 if (new_size > old_size) {
141 memset(&arena_ptr->buffer_ptr[arena_ptr->current_offset], 0, new_size - old_size);
142 }
143 return old_memory_ptr;
144 }
145
146 const size_t copy_size = old_size < new_size ? old_size : new_size;
147
148 void* new_memory = arena_allocate_aligned(arena_ptr, alignment, new_size);
149
150 // Copy across old memory to the new memory
151 memmove(new_memory, old_mem_buf, copy_size);
