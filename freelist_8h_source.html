82/* Get the prev size from header */
83static inline size_t freelist_header_prev_size(const struct freelist_header *header)
84{
85 return (header->__prev_block_size_w_freed_bit & ~(size_t)1);

88/* Check if the memory block under header is freed */
89static inline bool freelist_header_is_freed(const struct freelist_header *header)
90{
91 return header->__prev_block_size_w_freed_bit & 1;

94/* Get the pointer to previous header */
95static inline struct freelist_header *freelist_header_prev(struct freelist_header *header)
96{

99 return prev_size == 0 ? NULL : (struct freelist_header *)((char *)header - prev_size);

102/* Get the pointer to next header */
103static inline struct freelist_header *freelist_header_next(struct freelist_header *header,
104 const struct freelist *freelist_ptr)
105{
106 const char *next = (char *)header + header->curr_block_size;
107 const bool is_end = next >= (char *)freelist_ptr->buf_ptr + freelist_ptr->buf_len;

109 return is_end ? NULL : (struct freelist_header *)next;

120{

132/* Initialize the freelist */
133static inline void freelist_init(struct freelist *self, const size_t len, unsigned char backing_buf[len])
134{
135 assert(self);
136 assert(backing_buf);
137
138 const uintptr_t padding = calc_alignment_padding(alignof(struct freelist_header), (uintptr_t)&backing_buf[0]);
139
140 assert(len >= sizeof(struct freetree_node) + padding);
141
142 self->buf_ptr = &backing_buf[padding];
143 self->buf_len = len - padding;
144
145 freelist_deallocate_all(self);
146}

148static inline void *internal_freelist_init_block(struct freelist *self, char *block_ptr, struct freelist_header *next,
149 const size_t prev_size, const size_t block_size,
150 const size_t remaining_size)
151{
152 if (remaining_size > 0) {
153 struct freelist_header *remaining_header = (struct freelist_header *)(block_ptr + block_size);
154 *remaining_header = freelist_header_from(true, block_size, remaining_size);
155
156 if (next) {
157 *next = freelist_header_from(freelist_header_is_freed(next), remaining_size, next->curr_block_size);
158 }
159
160 if (freelist_header_should_be_in_freetree(remaining_header)) {
161 struct freetree_node *new_node = (struct freetree_node *)remaining_header;
162 freetree_node_init(new_node, *remaining_header);
163 freetree_insert_node(&self->rb_rootptr, new_node);
164 }
165 }
166 struct freelist_header *curr_header = (struct freelist_header *)block_ptr;
167 *curr_header = freelist_header_from(false, prev_size, block_size);
168
169 return (void *)((char *)curr_header + sizeof(struct freelist_header));
170}

172/* Get the pointer to a block of the freelist.*/
173static inline void *freelist_allocate(struct freelist *self, const size_t requested_size)
174{
175 assert(self);
176 assert(requested_size != 0);
177
178 size_t block_size = sizeof(struct freelist_header) + requested_size;
179 block_size = block_size >= sizeof(struct freetree_node) ? block_size : sizeof(struct freetree_node);
180 block_size = block_size + calc_alignment_padding(alignof(struct freetree_node), block_size);
181
182 struct freetree_node *node = internal_freetree_search_best_block(&self->rb_rootptr, block_size);
183
184 if (node == NULL) {
185 return NULL;
186 }
187
188 freetree_delete_node(&self->rb_rootptr, node);
189
190 self->buf_used += block_size;
191
192 return internal_freelist_init_block(self, (char *)node, freelist_header_next(&node->key, self),
193 freelist_header_prev_size(&node->key), block_size,
194 node->key.curr_block_size - block_size);
195}

197/* Deallocate a block from the freelist for further use. */
198static inline void freelist_deallocate(struct freelist *self, void *ptr)
199{
200 struct freelist_header *header = (struct freelist_header *)((char *)ptr - sizeof(struct freelist_header));
201
202 assert(!freelist_header_is_freed(header) && "double free detected!");
203
204 self->buf_used -= header->curr_block_size;
205
206 internal_freelist_coalescence(self, header);
207}

260static inline struct freetree_node *internal_freetree_search_best_block(struct freetree_node **rootptr_ptr,
261 const size_t block_size)
262{
263 assert(rootptr_ptr != NULL);
264
265 struct freetree_node *prev_ptr = NULL;
266 struct freetree_node *curr_ptr = *rootptr_ptr;
267
268 while (curr_ptr != NULL) {
269 if (block_size > curr_ptr->key.curr_block_size) {
270 curr_ptr = curr_ptr->right_ptr;
271 }
272 else {
273 prev_ptr = curr_ptr;
274 curr_ptr = curr_ptr->left_ptr;
275 }
276 }
277 return prev_ptr;
278}

280static inline void internal_freelist_coalescence(struct freelist *self, struct freelist_header *header)
281{
282 assert(self);
283 assert(header);
284
285 struct freelist_header *prev = freelist_header_prev(header);
286 struct freelist_header *next = freelist_header_next(header, self);
287
288 struct freelist_header header_new = *header;
289 struct freelist_header *header_addr = header;
290 struct freelist_header *header_next = next;
291
292 if (prev && freelist_header_is_freed(prev)) {
293 header_new.curr_block_size += prev->curr_block_size;
294 header_new.__prev_block_size_w_freed_bit = prev->__prev_block_size_w_freed_bit & ~(uintptr_t)1;
295 header_addr = prev;
296
297 if (freelist_header_should_be_in_freetree(prev)) {
298 freetree_delete_node(&self->rb_rootptr, (struct freetree_node *)prev);
299 }
300 }
301 if (next && freelist_header_is_freed(next)) {
302 header_new.curr_block_size += next->curr_block_size;
303 header_next = freelist_header_next(next, self);
304
305 if (freelist_header_should_be_in_freetree(next)) {
306 freetree_delete_node(&self->rb_rootptr, (struct freetree_node *)next);
307 }
308 }
309 if (header_next) {
310 *header_next = freelist_header_from(freelist_header_is_freed(header_next), header_new.curr_block_size,
311 header_next->curr_block_size);
312 }
313 header_new.__prev_block_size_w_freed_bit |= 1;
314 assert(freelist_header_should_be_in_freetree(&header_new));
315
316 struct freetree_node *node_addr = (struct freetree_node *)header_addr;
317 freetree_node_init(node_addr, header_new);
318 freetree_insert_node(&self->rb_rootptr, node_addr);
319}
