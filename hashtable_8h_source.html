445static inline bool JOIN(HASHTABLE_PREFIX, grow)(HASHTABLE_TYPE* hashtable_ptr, size_t new_capacity) {
446 assert(NULL != hashtable_ptr);
447
448 size_t old_capacity = hashtable_ptr->index_mask + 1;
449 if (!is_pow2(new_capacity) || new_capacity <= old_capacity || new_capacity > SIZE_MAX / sizeof(VALUE_TYPE)) {
450 return false;
451 }
452 HASHTABLE_SLOT_TYPE* new_slots = (HASHTABLE_SLOT_TYPE*)hashtable_ptr->allocator_ops.allocate_f_ptr(
453 hashtable_ptr->allocator_context_ptr, alignof(HASHTABLE_SLOT_TYPE), sizeof(HASHTABLE_SLOT_TYPE) * new_capacity);
454 if (new_slots == NULL) {
455 return false;
456 }
457
458 for (size_t i = 0; i < new_capacity; i++) {
459 new_slots[i].offset = EMPTY_HASHTABLE_SLOT_OFFSET;
460 }
461
462 size_t new_index_mask = new_capacity - 1;
463 for (size_t index = 0; index < old_capacity; (index)++) {
464 if (hashtable_ptr->slots[index].offset == EMPTY_HASHTABLE_SLOT_OFFSET) {
465 continue;
466 }
467 // insertion algorithm - modified for rehashing.
468 KEY_TYPE key = hashtable_ptr->slots[index].key;
469 VALUE_TYPE value = hashtable_ptr->slots[index].value;
470
471 size_t key_hash = HASH_FUNCTION(key);
472 size_t index = key_hash & new_index_mask;
473 HASHTABLE_SLOT_TYPE current_slot = {.offset = 0, .key = key, .value = value};
474
475 while (new_slots[index].offset != EMPTY_HASHTABLE_SLOT_OFFSET) {
476 if (current_slot.offset > new_slots[index].offset) {
477 HASHTABLE_SLOT_TYPE temp = new_slots[index];
478 new_slots[index] = current_slot;
479 current_slot = temp;
480 }
481 index = (index + 1) & new_index_mask;
482 current_slot.offset++;
483 }
484 new_slots[index] = current_slot;
485 }
486
487 hashtable_ptr->allocator_ops.deallocate_f_ptr(hashtable_ptr->allocator_context_ptr, hashtable_ptr->slots);
488 hashtable_ptr->slots = new_slots;
489 hashtable_ptr->index_mask = new_index_mask;
490
491 return true;
492}

508static inline bool JOIN(HASHTABLE_PREFIX, insert)(HASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key, VALUE_TYPE value) {
509 assert(NULL != hashtable_ptr);
510 assert(false == JOIN(HASHTABLE_PREFIX, contains_key)(hashtable_ptr, key));
511
512 size_t old_capacity = hashtable_ptr->index_mask + 1;
513 if (hashtable_ptr->count >= old_capacity / 2 && !JOIN(HASHTABLE_PREFIX, grow)(hashtable_ptr, 2 * old_capacity)) {
514 return false;
515 }

598static inline bool JOIN(HASHTABLE_PREFIX, delete)(HASHTABLE_TYPE* hashtable_ptr, KEY_TYPE key) {
599 assert(hashtable_ptr != NULL);
600
601 size_t key_hash = HASH_FUNCTION(key);
602 size_t index = key_hash & hashtable_ptr->index_mask;
603 size_t max_possible_offset = 0;
604
605 while (hashtable_ptr->slots[index].offset != EMPTY_HASHTABLE_SLOT_OFFSET &&
606 max_possible_offset <= hashtable_ptr->slots[index].offset) {
607 if (KEY_IS_EQUAL(hashtable_ptr->slots[index].key, key)) {
608 // mark as deleted:
609 hashtable_ptr->slots[index].offset = EMPTY_HASHTABLE_SLOT_OFFSET;
610 hashtable_ptr->count--;
611
612 // reduce offsets as much as possible by moving back offset elements:
613 size_t next_index = (index + 1) & hashtable_ptr->index_mask;
614 while (hashtable_ptr->slots[next_index].offset != EMPTY_HASHTABLE_SLOT_OFFSET &&
615 hashtable_ptr->slots[next_index].offset > 0) {
616 hashtable_ptr->slots[index] = hashtable_ptr->slots[next_index];
617 hashtable_ptr->slots[index].offset--;
618 hashtable_ptr->slots[next_index].offset = EMPTY_HASHTABLE_SLOT_OFFSET;
619
620 index = next_index;
621 next_index = (index + 1) & hashtable_ptr->index_mask;
622 }
623 return true;
624 }
625 index = (index + 1) & hashtable_ptr->index_mask;
626 max_possible_offset++;
627 }
628 return false;
629}
