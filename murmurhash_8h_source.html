38static inline uint32_t murmur3_32(const uint8_t* key_ptr, const size_t len, const uint32_t seed) {
39 uint32_t h = seed;
40 uint32_t k;
41
42 /* Read in groups of 4. */
43 for (size_t i = len >> 2; i; i--) {
44 // Here is a source of differing results across endiannesses.
45 // A swap here has no effects on hash properties though.
46 memcpy(&k, key_ptr, sizeof(uint32_t));
47 key_ptr += sizeof(uint32_t);
48 h ^= murmur_32_scramble(k);
49 h = (h << 13) | (h >> 19);
50 h = h * 5 + 0xe6546b64;
51 }
52
53 /* Read the rest. */
54 k = 0;
55 for (size_t i = len & 3; i; i--) {
56 k <<= 8;
57 k |= key_ptr[i - 1];
58 }
59
60 // A swap is *not* necessary here because the preceding loop already
61 // places the low bytes in the low places according to whatever endianness
62 // we use. Swaps only apply when the memory is copied in a chunk.
63 h ^= murmur_32_scramble(k);
64
65 /* Finalize. */
66 h ^= len;
67 h ^= h >> 16;
68 h *= 0x85ebca6b;
69 h ^= h >> 13;
70 h *= 0xc2b2ae35;
71 h ^= h >> 16;
72 return h;
