42static inline uint32_t murmur_32_scramble(uint32_t k) {
43 k *= 0xcc9e2d51;
44 k = (k << 15) | (k >> 17);
45 k *= 0x1b873593;
46 return k;

58static inline uint32_t murmur3_32(const uint8_t* key_ptr, const size_t len, const uint32_t seed) {
59 uint32_t h = seed;
60 uint32_t k;
61
62 /* Read in groups of 4. */
63 for (size_t i = len >> 2; i; i--) {
64 // Here is a source of differing results across endiannesses.
65 // A swap here has no effects on hash properties though.
66 memcpy(&k, key_ptr, sizeof(uint32_t));
67 key_ptr += sizeof(uint32_t);
68 h ^= murmur_32_scramble(k);
69 h = (h << 13) | (h >> 19);
70 h = h * 5 + 0xe6546b64;
71 }
72
73 /* Read the rest. */
74 k = 0;
75 for (size_t i = len & 3; i; i--) {
76 k <<= 8;
77 k |= key_ptr[i - 1];
78 }
79
80 // A swap is *not* necessary here because the preceding loop already
81 // places the low bytes in the low places according to whatever endianness
82 // we use. Swaps only apply when the memory is copied in a chunk.
83 h ^= murmur_32_scramble(k);
84
85 /* Finalize. */
86 h ^= len;
87 h ^= h >> 16;
88 h *= 0x85ebca6b;
89 h ^= h >> 13;
90 h *= 0xc2b2ae35;
91 h ^= h >> 16;
92 return h;
93}
