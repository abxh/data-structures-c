135static inline FQUEUE_TYPE* JOIN(FQUEUE_NAME, create)(const uint32_t capacity)
136{
137 if (capacity == 0 || capacity > UINT32_MAX / 4) {
138 return NULL;
139 }
140 const uint32_t capacity_new = round_up_pow2_32(capacity);
141
142 FQUEUE_TYPE* queue_ptr = (FQUEUE_TYPE*)calloc(1, offsetof(FQUEUE_TYPE, values) + capacity_new * sizeof(VALUE_TYPE));
143
144 if (!queue_ptr) {
145 return NULL;
146 }
147
148 queue_ptr->begin_index = queue_ptr->end_index = 0;
149 queue_ptr->count = 0;
150 queue_ptr->capacity = capacity_new;
151
152 return queue_ptr;
153}

263static inline bool JOIN(FQUEUE_NAME, enqueue)(FQUEUE_TYPE* queue_ptr, const VALUE_TYPE value)
264{
265 assert(queue_ptr != NULL);
266 assert(!FQUEUE_IS_FULL(queue_ptr));
267
268 const uint32_t index_mask = (queue_ptr->capacity - 1);
269
270 queue_ptr->values[queue_ptr->end_index] = value;
271 queue_ptr->end_index++;
272 queue_ptr->end_index &= index_mask;
273 queue_ptr->count++;
274
275 return true;

318static inline void JOIN(FQUEUE_NAME, copy)(FQUEUE_TYPE* restrict dest_queue_ptr, const FQUEUE_TYPE* restrict src_queue_ptr)
319{
320 assert(src_queue_ptr != NULL);
321 assert(dest_queue_ptr != NULL);
322 assert(src_queue_ptr->count <= dest_queue_ptr->capacity);
323 assert(FQUEUE_IS_EMPTY(dest_queue_ptr));
324
325 const uint32_t src_begin_index = src_queue_ptr->begin_index;
326 const uint32_t src_index_mask = src_queue_ptr->capacity - 1;
327
328 for (uint32_t i = 0; i < src_queue_ptr->count; i++) {
329 dest_queue_ptr->values[i] = src_queue_ptr->values[(src_begin_index + i) & src_index_mask];
330 }
331
332 dest_queue_ptr->count = src_queue_ptr->count;
333 dest_queue_ptr->begin_index = 0;
334 dest_queue_ptr->end_index = src_queue_ptr->count;
335}
