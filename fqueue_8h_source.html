137static inline FQUEUE_TYPE* JOIN(FQUEUE_NAME, create)(const uint32_t capacity)
138{
139 if (capacity == 0 || capacity > UINT32_MAX / 4) {
140 return NULL;
141 }
142 const uint32_t capacity_new = round_up_pow2_32(capacity);
143
144 FQUEUE_TYPE* queue_ptr = (FQUEUE_TYPE*)malloc(offsetof(FQUEUE_TYPE, values) + capacity_new * sizeof(VALUE_TYPE));
145 if (!queue_ptr) {
146 return NULL;
147 }
148 memset(queue_ptr, 0, offsetof(FQUEUE_TYPE, values) + capacity * sizeof(VALUE_TYPE));
149
150 queue_ptr->begin_index = queue_ptr->end_index = 0;
151 queue_ptr->count = 0;
152 queue_ptr->capacity = capacity_new;
153
154 return queue_ptr;
155}

298static inline bool JOIN(FQUEUE_NAME, enqueue)(FQUEUE_TYPE* queue_ptr, const VALUE_TYPE value)
299{
300 assert(queue_ptr != NULL);
301 assert(!FQUEUE_IS_FULL(queue_ptr));
302
303 const int index_mask = (queue_ptr->capacity - 1);
304
305 queue_ptr->values[queue_ptr->end_index] = value;
306 queue_ptr->end_index++;
307 queue_ptr->end_index &= index_mask;
308 queue_ptr->count++;

366static inline void JOIN(FQUEUE_NAME, copy)(FQUEUE_TYPE* restrict dest_queue_ptr, const FQUEUE_TYPE* restrict src_queue_ptr)
367{
368 assert(src_queue_ptr != NULL);
369 assert(dest_queue_ptr != NULL);
370 assert(src_queue_ptr->count <= dest_queue_ptr->capacity);
371 assert(FQUEUE_IS_EMPTY(dest_queue_ptr));
372
373 const uint32_t src_begin_index = src_queue_ptr->begin_index;
374 const uint32_t src_index_mask = src_queue_ptr->capacity - 1;
375
376 for (uint32_t i = 0; i < src_queue_ptr->count; i++) {
377 dest_queue_ptr->values[i] = src_queue_ptr->values[(src_begin_index + i) & src_index_mask];
378 }
379
380 dest_queue_ptr->count = src_queue_ptr->count;
381 dest_queue_ptr->begin_index = 0;
382 dest_queue_ptr->end_index = src_queue_ptr->count;
383}
