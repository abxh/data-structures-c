147static inline FQUEUE_TYPE* JOIN(FQUEUE_NAME, create)(const size_t capacity) {
148 if (capacity == 0 || capacity > UINT32_MAX / 4) {
149 return NULL;
150 }
151 const size_t capacity_new = round_up_pow2(capacity);
152
153 FQUEUE_TYPE* queue_ptr = (FQUEUE_TYPE*)malloc(offsetof(FQUEUE_TYPE, values) + capacity_new * sizeof(VALUE_TYPE));
154 if (!queue_ptr) {
155 return NULL;
156 }
157 memset(queue_ptr, 0, offsetof(FQUEUE_TYPE, values) + capacity * sizeof(VALUE_TYPE));
158
159 queue_ptr->begin_index = queue_ptr->end_index = 0;
160 queue_ptr->count = 0;
161 queue_ptr->capacity = capacity_new;
162
163 return queue_ptr;
164}

293static inline bool JOIN(FQUEUE_NAME, enqueue)(FQUEUE_TYPE* queue_ptr, const VALUE_TYPE value) {
294 assert(queue_ptr != NULL);
295 assert(!FQUEUE_IS_FULL(queue_ptr));
296
297 const int index_mask = (queue_ptr->capacity - 1);

299 queue_ptr->values[queue_ptr->end_index] = value;

357static inline void JOIN(FQUEUE_NAME, copy)(FQUEUE_TYPE* restrict dest_queue_ptr, const FQUEUE_TYPE* restrict src_queue_ptr) {
358 assert(src_queue_ptr != NULL);
359 assert(dest_queue_ptr != NULL);
360 assert(src_queue_ptr->count <= dest_queue_ptr->capacity);
361 assert(FQUEUE_IS_EMPTY(dest_queue_ptr));
362
363 const size_t src_begin_index = src_queue_ptr->begin_index;
364 const size_t src_index_mask = src_queue_ptr->capacity - 1;
365
366 for (size_t i = 0; i < src_queue_ptr->count; i++) {
367 dest_queue_ptr->values[i] = src_queue_ptr->values[(src_begin_index + i) & src_index_mask];
368 }
369
370 dest_queue_ptr->count = src_queue_ptr->count;
371 dest_queue_ptr->begin_index = 0;
372 dest_queue_ptr->end_index = src_queue_ptr->count;
373}
