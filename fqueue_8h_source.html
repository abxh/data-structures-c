145static inline FQUEUE_TYPE* JOIN(FQUEUE_NAME, create)(const uint32_t capacity)
146{
147 if (capacity == 0 || capacity > UINT32_MAX / 4) {
148 return NULL;
149 }
150 const uint32_t capacity_new = round_up_pow2_32(capacity);
151
152 FQUEUE_TYPE* queue_ptr = (FQUEUE_TYPE*)malloc(offsetof(FQUEUE_TYPE, values) + capacity_new * sizeof(VALUE_TYPE));
153 if (!queue_ptr) {
154 return NULL;
155 }
156 memset(queue_ptr, 0, offsetof(FQUEUE_TYPE, values) + capacity * sizeof(VALUE_TYPE));
157
158 queue_ptr->begin_index = queue_ptr->end_index = 0;
159 queue_ptr->count = 0;
160 queue_ptr->capacity = capacity_new;
161
162 return queue_ptr;
163}

299static inline bool JOIN(FQUEUE_NAME, enqueue)(FQUEUE_TYPE* queue_ptr, const VALUE_TYPE value)
300{
301 assert(queue_ptr != NULL);
302 assert(!FQUEUE_IS_FULL(queue_ptr));
303
304 const int index_mask = (queue_ptr->capacity - 1);
305
306 queue_ptr->values[queue_ptr->end_index] = value;
307 queue_ptr->end_index++;
308 queue_ptr->end_index &= index_mask;
309 queue_ptr->count++;
310
311 return true;
312}

366static inline void JOIN(FQUEUE_NAME, copy)(FQUEUE_TYPE* restrict dest_queue_ptr, const FQUEUE_TYPE* restrict src_queue_ptr)
367{
368 assert(src_queue_ptr != NULL);
369 assert(dest_queue_ptr != NULL);
370 assert(src_queue_ptr->count <= dest_queue_ptr->capacity);
371 assert(FQUEUE_IS_EMPTY(dest_queue_ptr));
372
373 const uint32_t src_begin_index = src_queue_ptr->begin_index;
374 const uint32_t src_index_mask = src_queue_ptr->capacity - 1;
375
376 for (uint32_t i = 0; i < src_queue_ptr->count; i++) {
377 dest_queue_ptr->values[i] = src_queue_ptr->values[(src_begin_index + i) & src_index_mask];
378 }
379
380 dest_queue_ptr->count = src_queue_ptr->count;
381 dest_queue_ptr->begin_index = 0;
382 dest_queue_ptr->end_index = src_queue_ptr->count;
383}
