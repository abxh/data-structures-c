147static inline FQUEUE_TYPE* JOIN(FQUEUE_NAME, create)(const uint32_t capacity)
148{
149 if (capacity == 0 || capacity > UINT32_MAX / 4) {
150 return NULL;
151 }
152 const uint32_t capacity_new = round_up_pow2_32(capacity);
153
154 FQUEUE_TYPE* queue_ptr = (FQUEUE_TYPE*)malloc(offsetof(FQUEUE_TYPE, values) + capacity_new * sizeof(VALUE_TYPE));
155 if (!queue_ptr) {
156 return NULL;
157 }
158 memset(queue_ptr, 0, offsetof(FQUEUE_TYPE, values) + capacity * sizeof(VALUE_TYPE));
159
160 queue_ptr->begin_index = queue_ptr->end_index = 0;
161 queue_ptr->count = 0;
162 queue_ptr->capacity = capacity_new;
163
164 return queue_ptr;
165}

301static inline bool JOIN(FQUEUE_NAME, enqueue)(FQUEUE_TYPE* queue_ptr, const VALUE_TYPE value)

303 assert(queue_ptr != NULL);

368static inline void JOIN(FQUEUE_NAME, copy)(FQUEUE_TYPE* restrict dest_queue_ptr, const FQUEUE_TYPE* restrict src_queue_ptr)
369{
370 assert(src_queue_ptr != NULL);
371 assert(dest_queue_ptr != NULL);
372 assert(src_queue_ptr->count <= dest_queue_ptr->capacity);
373 assert(FQUEUE_IS_EMPTY(dest_queue_ptr));
374
375 const uint32_t src_begin_index = src_queue_ptr->begin_index;
376 const uint32_t src_index_mask = src_queue_ptr->capacity - 1;
377
378 for (uint32_t i = 0; i < src_queue_ptr->count; i++) {
379 dest_queue_ptr->values[i] = src_queue_ptr->values[(src_begin_index + i) & src_index_mask];
380 }
381
382 dest_queue_ptr->count = src_queue_ptr->count;
383 dest_queue_ptr->begin_index = 0;
384 dest_queue_ptr->end_index = src_queue_ptr->count;
385}
