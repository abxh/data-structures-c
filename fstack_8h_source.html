150static inline FSTACK_TYPE* JOIN(FSTACK_NAME, create)(const uint32_t capacity)
151{
152 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FSTACK_TYPE, values)) / sizeof(VALUE_TYPE)) {
153 return NULL;
154 }
155 FSTACK_TYPE* stack_ptr = (FSTACK_TYPE*)malloc(offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
156
157 if (!stack_ptr) {
158 return NULL;
159 }
160 memset(stack_ptr, 0, offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
161
162 stack_ptr->count = 0;
163 stack_ptr->capacity = capacity;
164
165 return stack_ptr;
166}

344static inline void JOIN(FSTACK_NAME, copy)(FSTACK_TYPE* restrict dest_stack_ptr, const FSTACK_TYPE* restrict src_stack_ptr)

347 assert(dest_stack_ptr != NULL);
348 assert(src_stack_ptr->count <= dest_stack_ptr->capacity);
349 assert(FSTACK_IS_EMPTY(dest_stack_ptr));
