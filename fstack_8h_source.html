149static inline FSTACK_TYPE* JOIN(FSTACK_NAME, create)(const uint32_t capacity)
150{
151 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FSTACK_TYPE, values)) / sizeof(VALUE_TYPE)) {
152 return NULL;
153 }
154 FSTACK_TYPE* stack_ptr = (FSTACK_TYPE*)malloc(offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
155
156 if (!stack_ptr) {
157 return NULL;
158 }
159 memset(stack_ptr, 0, offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
160
161 stack_ptr->count = 0;
162 stack_ptr->capacity = capacity;
163
164 return stack_ptr;
165}

351static inline void JOIN(FSTACK_NAME, copy)(FSTACK_TYPE* restrict dest_stack_ptr, const FSTACK_TYPE* restrict src_stack_ptr)
352{
353 assert(src_stack_ptr != NULL);
354 assert(dest_stack_ptr != NULL);
355 assert(src_stack_ptr->count <= dest_stack_ptr->capacity);
356 assert(FSTACK_IS_EMPTY(dest_stack_ptr));
357
358 for (uint32_t i = 0; i < src_stack_ptr->count; i++) {
359 dest_stack_ptr->values[i] = src_stack_ptr->values[i];
360 }
361 dest_stack_ptr->count = src_stack_ptr->count;
362}
