140static inline FSTACK_TYPE* JOIN(FSTACK_NAME, create)(const uint32_t capacity)
141{
142 if (capacity == 0 || capacity > (UINT32_MAX - offsetof(FSTACK_TYPE, values)) / sizeof(VALUE_TYPE)) {
143 return NULL;
144 }
145 FSTACK_TYPE* stack_ptr = (FSTACK_TYPE*)malloc(offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
146
147 if (!stack_ptr) {
148 return NULL;
149 }
150 memset(stack_ptr, 0, offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
151
152 stack_ptr->count = 0;
153 stack_ptr->capacity = capacity;
154
155 return stack_ptr;
156}

342static inline void JOIN(FSTACK_NAME, copy)(FSTACK_TYPE* restrict dest_stack_ptr, const FSTACK_TYPE* restrict src_stack_ptr)
343{
344 assert(src_stack_ptr != NULL);
345 assert(dest_stack_ptr != NULL);
346 assert(src_stack_ptr->count <= dest_stack_ptr->capacity);
347 assert(FSTACK_IS_EMPTY(dest_stack_ptr));
348
349 for (uint32_t i = 0; i < src_stack_ptr->count; i++) {
350 dest_stack_ptr->values[i] = src_stack_ptr->values[i];
351 }
352 dest_stack_ptr->count = src_stack_ptr->count;
