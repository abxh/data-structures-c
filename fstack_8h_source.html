150static inline FSTACK_TYPE* JOIN(FSTACK_NAME, create)(const size_t capacity) {
151 if (capacity == 0 || capacity > (SIZE_MAX - offsetof(FSTACK_TYPE, values)) / sizeof(VALUE_TYPE)) {
152 return NULL;
153 }
154 FSTACK_TYPE* stack_ptr = (FSTACK_TYPE*)malloc(offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
155
156 if (!stack_ptr) {
157 return NULL;
158 }
159 memset(stack_ptr, 0, offsetof(FSTACK_TYPE, values) + capacity * sizeof(VALUE_TYPE));
160
161 stack_ptr->count = 0;
162 stack_ptr->capacity = capacity;
163
164 return stack_ptr;
165}

333static inline void JOIN(FSTACK_NAME, copy)(FSTACK_TYPE* restrict dest_stack_ptr, const FSTACK_TYPE* restrict src_stack_ptr) {
334 assert(src_stack_ptr != NULL);
335 assert(dest_stack_ptr != NULL);
336 assert(src_stack_ptr->count <= dest_stack_ptr->capacity);
337 assert(FSTACK_IS_EMPTY(dest_stack_ptr));

339 for (size_t i = 0; i < src_stack_ptr->count; i++) {
