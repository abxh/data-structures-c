173static inline FSTACK_TYPE *JOIN(FSTACK_NAME, create)(const uint32_t capacity)
174{
175 if (capacity == 0 || fstack_calc_sizeof_overflows(FSTACK_NAME, capacity)) {
176 return NULL;
177 }

179 const uint32_t size = fstack_calc_sizeof(FSTACK_NAME, capacity);

345 assert(src_ptr->count <= dest_ptr->capacity);
