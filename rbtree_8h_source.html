333static inline bool JOIN(RBTREE_NAME, contains_key)(RBTREE_NODE_TYPE **rootptr_ptr, const KEY_TYPE key)
334{
335 assert(rootptr_ptr != NULL);
336
337 RBTREE_NODE_TYPE *node_ptr = *rootptr_ptr;
338
339 while (node_ptr != NULL) {
340 const bool is_strictly_less = KEY_IS_STRICTLY_LESS(key, node_ptr->key);
341 const bool is_strictly_greater = KEY_IS_STRICTLY_LESS(node_ptr->key, key);
342 const bool is_equal = !is_strictly_less && !is_strictly_greater;
343
344 if (is_equal) {
345 return true;
346 }
347 else if (is_strictly_less) {
348 node_ptr = node_ptr->left_ptr;
349 }
350 else {
351 node_ptr = node_ptr->right_ptr;
352 }
353 }
354 return false;
355}

395static inline void JOIN(RBTREE_NAME, insert_node)(RBTREE_NODE_TYPE **rootptr_ptr, RBTREE_NODE_TYPE *node_ptr)
396{
397 assert(rootptr_ptr != NULL);
398 assert(node_ptr != NULL);
399#ifndef ALLOW_DUPLICATES
400 assert(RBTREE_CONTAINS_KEY(rootptr_ptr, node_ptr->key) == false);
401#endif
402
403 RBTREE_NODE_TYPE *parent_ptr = NULL;
404 RBTREE_NODE_TYPE *current_ptr = *rootptr_ptr;
405
406 while (current_ptr != NULL) {
407 const bool is_strictly_greater = KEY_IS_STRICTLY_LESS(current_ptr->key, node_ptr->key);
408
409 parent_ptr = current_ptr;
410
411 if (is_strictly_greater) {
412 current_ptr = current_ptr->right_ptr;
413 }
414 else {
415 current_ptr = current_ptr->left_ptr;
416 }
417 }
418
419 node_ptr->left_ptr = node_ptr->right_ptr = NULL;
420 RBTREE_NODE_SET_PARENT_PTR(node_ptr, parent_ptr);
421 RBTREE_NODE_SET_COLOR_TO_RED(node_ptr);
422
423 if (parent_ptr == NULL) {
424 *rootptr_ptr = node_ptr;
425 }
426 else {
427 const int dir = KEY_IS_STRICTLY_LESS(parent_ptr->key, node_ptr->key) ? 1 : 0;
428
429 parent_ptr->child_ptrs[dir] = node_ptr;
430
431 RBTREE_INSERT_FIXUP(rootptr_ptr, node_ptr);
432 }
433}

443static inline RBTREE_NODE_TYPE *JOIN(RBTREE_NAME, delete_node)(RBTREE_NODE_TYPE **rootptr_ptr,
444 RBTREE_NODE_TYPE *node_ptr)
445{
446 assert(rootptr_ptr != NULL);
447 assert(node_ptr != NULL);
448
449 if (node_ptr->left_ptr == NULL && node_ptr->right_ptr == NULL) {
450 if (node_ptr == *rootptr_ptr || RBTREE_NODE_IS_RED(node_ptr)) {
451 RBTREE_NODE_TRANSPLANT(rootptr_ptr, node_ptr, NULL);
452 }
453 else {
454 RBTREE_NODE_TYPE *const parent_ptr = RBTREE_NODE_GET_PARENT_PTR(node_ptr);
455
456 const int dir = RBTREE_CHILD_DIR(node_ptr) ? 1 : 0;
457
458 RBTREE_NODE_TRANSPLANT(rootptr_ptr, node_ptr, NULL);
459
460 RBTREE_DELETE_FIXUP(rootptr_ptr, parent_ptr, dir);
461 }
462 }
463 else if (node_ptr->left_ptr == NULL || node_ptr->right_ptr == NULL) {
464 const int dir = node_ptr->left_ptr == NULL;
465
466 assert(RBTREE_NODE_IS_BLACK(node_ptr));
467 assert(RBTREE_NODE_IS_RED(node_ptr->child_ptrs[dir]));
468
469 RBTREE_NODE_SET_COLOR_TO_BLACK(node_ptr->child_ptrs[dir]);
470
471 RBTREE_NODE_TRANSPLANT(rootptr_ptr, node_ptr, node_ptr->child_ptrs[dir]);
472 }
473 else {
474 RBTREE_NODE_TYPE *successor_ptr = node_ptr->right_ptr;
475
476 while (successor_ptr->left_ptr != NULL) {
477 successor_ptr = successor_ptr->left_ptr;
478 }
479
480 const bool prev_successor_black = RBTREE_NODE_IS_BLACK(successor_ptr);
481
482 const int prev_successor_dir = RBTREE_CHILD_DIR(successor_ptr);
483
484 RBTREE_NODE_TYPE *const prev_successor_parent_ptr = RBTREE_NODE_GET_PARENT_PTR(successor_ptr);
485
486 RBTREE_NODE_TYPE *const prev_successor_child_ptr = successor_ptr->right_ptr;
487
488 {
489 if (RBTREE_NODE_GET_PARENT_PTR(successor_ptr) != node_ptr) {
490 RBTREE_NODE_TRANSPLANT(rootptr_ptr, successor_ptr, successor_ptr->right_ptr);
491
492 successor_ptr->right_ptr = node_ptr->right_ptr;
493 RBTREE_NODE_SET_PARENT_PTR(node_ptr->right_ptr, successor_ptr);
494 }
495 RBTREE_NODE_TRANSPLANT(rootptr_ptr, node_ptr, successor_ptr);
496
497 successor_ptr->left_ptr = node_ptr->left_ptr;
498 RBTREE_NODE_SET_PARENT_PTR(node_ptr->left_ptr, successor_ptr);
499
500 RBTREE_NODE_SET_COLOR_TO_COLOR_OF_OTHER(successor_ptr, node_ptr);
501 }
502
503 if (prev_successor_child_ptr != NULL) {
504 assert(prev_successor_black);
505 assert(RBTREE_NODE_IS_RED(prev_successor_child_ptr));
506
507 RBTREE_NODE_SET_COLOR_TO_BLACK(prev_successor_child_ptr);
508 }
509 else if (prev_successor_child_ptr == NULL && prev_successor_black) {
510 RBTREE_NODE_TYPE *actual_successor_ptr =
511 (prev_successor_parent_ptr == node_ptr) ? successor_ptr : prev_successor_parent_ptr;
512
513 RBTREE_DELETE_FIXUP(rootptr_ptr, actual_successor_ptr, prev_successor_dir);
514 }
515 }
516
517 node_ptr->left_ptr = node_ptr->right_ptr = NULL;
518 RBTREE_NODE_SET_PARENT_PTR(node_ptr, NULL);
519
520 return node_ptr;
521}

525static inline RBTREE_NODE_TYPE *JOIN(internal, JOIN(RBTREE_NAME, rotate_dir))(RBTREE_NODE_TYPE **rootptr_ptr,
526 RBTREE_NODE_TYPE *P, const int dir)
527{
528 assert(rootptr_ptr != NULL);
529 assert(P != NULL);
530 assert(dir == 0 || dir == 1);
531
532 /* Right rotate around P:
533 P S
534 / \ / \
535 S T --> .. P
536 / \ / \ / \
537 .. C .. .. C T
538 / \ / \ / \
539 .. .. .. .. .. ..
540
541 Left rotate around P:
542 P S
543 / \ / \
544 T S P ..
545 / \ / \ --> / \
546 .. .. C .. T C
547 / \ / \ / \
548 .. .. .. .. .. ..
549 */
550
551 RBTREE_NODE_TYPE *G = RBTREE_NODE_GET_PARENT_PTR(P);
552 RBTREE_NODE_TYPE *S = P->child_ptrs[1 - dir];
553 assert(S != NULL);
554 RBTREE_NODE_TYPE *C = S->child_ptrs[dir];
555
556 P->child_ptrs[1 - dir] = C;
557 if (C != NULL) {
558 RBTREE_NODE_SET_PARENT_PTR(C, P);
559 }
560
561 S->child_ptrs[dir] = P;
562 RBTREE_NODE_SET_PARENT_PTR(P, S);
563
564 RBTREE_NODE_SET_PARENT_PTR(S, G);
565 if (G != NULL) {
566 G->child_ptrs[P == G->left_ptr ? 0 : 1] = S;
567 }
568 else {
569 *rootptr_ptr = S;
570 }
571 return S;
572}

574static inline void JOIN(internal, JOIN(RBTREE_NAME, insert_fixup))(RBTREE_NODE_TYPE **rootptr_ptr, RBTREE_NODE_TYPE *N)
575{
576 assert(rootptr_ptr != NULL);
577 assert(N != NULL);
578 assert(RBTREE_NODE_IS_RED(N));
579 assert(N != *rootptr_ptr);
580
581 RBTREE_NODE_TYPE *P = RBTREE_NODE_GET_PARENT_PTR(N); // parent
582 RBTREE_NODE_TYPE *G = NULL; // grandparent
583 RBTREE_NODE_TYPE *U = NULL; // uncle
584
585 do {
586 if (RBTREE_NODE_IS_BLACK(P)) {
587 return;
588 }
589 if ((G = RBTREE_NODE_GET_PARENT_PTR(P)) == NULL) {
590 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
591 return;
592 }
593 const int dir = RBTREE_CHILD_DIR(P);
594 U = G->child_ptrs[1 - dir];
595
596 if (U == NULL || RBTREE_NODE_IS_BLACK(U)) {
597 if (N == P->child_ptrs[1 - dir]) {
598 RBTREE_ROTATE_DIR(rootptr_ptr, P, dir);
599 N = P;
600 P = G->child_ptrs[dir];
601 }
602 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
603 RBTREE_NODE_SET_COLOR_TO_RED(G);
604 RBTREE_ROTATE_DIR(rootptr_ptr, G, 1 - dir);
605 return;
606 }
607
608 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
609 RBTREE_NODE_SET_COLOR_TO_BLACK(U);
610 RBTREE_NODE_SET_COLOR_TO_RED(G);
611
612 N = G;
613 P = RBTREE_NODE_GET_PARENT_PTR(N);
614 } while (P != NULL);
615}

638static inline void JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(RBTREE_NODE_TYPE **rootptr_ptr, RBTREE_NODE_TYPE *P,
639 int dir)
640{
641 assert(rootptr_ptr != NULL);
642 assert(P != NULL);
643 assert(dir == 0 || dir == 1);
644
645 RBTREE_NODE_TYPE *N = NULL; // node (temp)
646 RBTREE_NODE_TYPE *S = NULL; // sibling
647 RBTREE_NODE_TYPE *C = NULL; // close nephew
648 RBTREE_NODE_TYPE *D = NULL; // distant nephew
649
650 do {
651 S = P->child_ptrs[1 - dir];
652 D = S->child_ptrs[1 - dir];
653 C = S->child_ptrs[dir];
654
655 if (RBTREE_NODE_IS_RED(S)) {
656 goto Case_3;
657 }
658 if (D != NULL && RBTREE_NODE_IS_RED(D)) {
659 goto Case_6;
660 }
661 if (C != NULL && RBTREE_NODE_IS_RED(C)) {
662 goto Case_5;
663 }
664 if (RBTREE_NODE_IS_RED(P)) {
665 goto Case_4;
666 }
667 RBTREE_NODE_SET_COLOR_TO_RED(S);
668
669 N = P;
670 P = RBTREE_NODE_GET_PARENT_PTR(N);
671 } while (P != NULL && (dir = RBTREE_CHILD_DIR(N), true));
672
673 return;
674Case_3:
675 RBTREE_ROTATE_DIR(rootptr_ptr, P, dir);
676 RBTREE_NODE_SET_COLOR_TO_RED(P);
677 RBTREE_NODE_SET_COLOR_TO_BLACK(S);
678 S = C;
679 D = S->child_ptrs[1 - dir];
680 if (D != NULL && RBTREE_NODE_IS_RED(D)) {
681 goto Case_6;
682 }
683 C = S->child_ptrs[dir];
684 if (C != NULL && RBTREE_NODE_IS_RED(C)) {
685 goto Case_5;
686 }
687Case_4:
688 RBTREE_NODE_SET_COLOR_TO_RED(S);
689 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
690 return;
691Case_5:
692 RBTREE_ROTATE_DIR(rootptr_ptr, S, 1 - dir);
693 RBTREE_NODE_SET_COLOR_TO_RED(S);
694 RBTREE_NODE_SET_COLOR_TO_BLACK(C);
695 D = S;
696 S = C;
697Case_6:
698 RBTREE_ROTATE_DIR(rootptr_ptr, P, dir);
699 RBTREE_NODE_SET_COLOR_TO_COLOR_OF_OTHER(S, P);
700 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
701 RBTREE_NODE_SET_COLOR_TO_BLACK(D);
702 return;
703}
