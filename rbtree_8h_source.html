302static inline bool JOIN(RBTREE_NAME, contains_key)(const RBTREE_TYPE* rbtree_ptr, const KEY_TYPE key)
303{
304 assert(rbtree_ptr != NULL);
305
306 RBTREE_NODE_TYPE* node_ptr = rbtree_ptr->root_ptr;
307
308 while (node_ptr != NULL) {
309 const bool is_strictly_less = KEY_IS_STRICTLY_LESS(key, node_ptr->key);
310 const bool is_strictly_greater = KEY_IS_STRICTLY_LESS(node_ptr->key, key);
311
312 if (!is_strictly_less && !is_strictly_greater) {
313 return true;
314 }
315 else if (is_strictly_less) {
316 node_ptr = node_ptr->left_ptr;
317 }
318 else {
319 node_ptr = node_ptr->right_ptr;
320 }
321 }
322 return false;
323}

360// rotate a subtree around a given subtree root node and direction (0: left or 1: right). returns the new subtree root
361static inline RBTREE_NODE_TYPE* JOIN(internal, JOIN(RBTREE_NAME, rotate_dir))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* P,
362 const uint8_t dir)
363{
364 assert(rbtree_ptr != NULL);
365 assert(P != NULL);
366 assert(dir == 0 || dir == 1);
367
368 /* Right rotate around P:
369 P S
370 / \ / \
371 S T --> .. P
372 / \ / \ ⟳ / \
373 .. C .. .. C T
374 / \ / \ / \
375 .. .. .. .. .. ..
376
377 Left rotate around P:
378 P S
379 / \ / \
380 T S P ..
381 / \ / \ --> / \
382 .. .. C .. ↺ T C
383 / \ / \ / \
384 .. .. .. .. .. ..
385 */
386
387 RBTREE_NODE_TYPE* G = RBTREE_NODE_GET_PARENT_PTR(P);
388 RBTREE_NODE_TYPE* S = P->child_ptrs[1 - dir];
389 assert(S != NULL);
390 RBTREE_NODE_TYPE* C = S->child_ptrs[dir];
391
392 P->child_ptrs[1 - dir] = C;
393 if (C != NULL) {
394 RBTREE_NODE_SET_PARENT_PTR(C, P);
395 }
396
397 S->child_ptrs[dir] = P;
398 RBTREE_NODE_SET_PARENT_PTR(P, S);
399
400 RBTREE_NODE_SET_PARENT_PTR(S, G);
401 if (G != NULL) {
402 G->child_ptrs[P == G->left_ptr ? 0 : 1] = S;
403 }
404 else {
405 rbtree_ptr->root_ptr = S;
406 }
407 return S;
408}

410// rebalance tree after insert. see explanation in the sources linked above.
411static inline void JOIN(internal, JOIN(RBTREE_NAME, insert_fixup))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* N)
412{
413 assert(rbtree_ptr != NULL);
414 assert(N != NULL);
415 assert(RBTREE_NODE_IS_RED(N));
416 assert(N != rbtree_ptr->root_ptr);
417
418 RBTREE_NODE_TYPE* P = RBTREE_NODE_GET_PARENT_PTR(N); // parent
419 RBTREE_NODE_TYPE* G = NULL; // grandparent
420 RBTREE_NODE_TYPE* U = NULL; // uncle
421
422 do {
423 if (RBTREE_NODE_IS_BLACK(P)) {
424 return;
425 }
426 if ((G = RBTREE_NODE_GET_PARENT_PTR(P)) == NULL) {
427 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
428 return;
429 }
430 const uint8_t dir = RBTREE_CHILD_DIR(P);
431 U = G->child_ptrs[1 - dir];
432
433 if (U == NULL || RBTREE_NODE_IS_BLACK(U)) {
434 if (N == P->child_ptrs[1 - dir]) {
435 RBTREE_ROTATE_DIR(rbtree_ptr, P, dir);
436 N = P;
437 P = G->child_ptrs[dir];
438 }
439 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
440 RBTREE_NODE_SET_COLOR_TO_RED(G);
441 RBTREE_ROTATE_DIR(rbtree_ptr, G, 1 - dir);
442 return;
443 }
444
445 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
446 RBTREE_NODE_SET_COLOR_TO_BLACK(U);
447 RBTREE_NODE_SET_COLOR_TO_RED(G);
448
449 N = G;
450 P = RBTREE_NODE_GET_PARENT_PTR(N);
451 } while (P != NULL);
452}

466static inline void JOIN(RBTREE_NAME, insert_node)(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* node_ptr)
467{
468 assert(rbtree_ptr != NULL);
469 assert(node_ptr != NULL);
470 assert(RBTREE_CONTAINS_KEY(rbtree_ptr, node_ptr->key) == false);
471
472 RBTREE_NODE_TYPE* parent_ptr = NULL;
473 RBTREE_NODE_TYPE* current_ptr = rbtree_ptr->root_ptr;
474
475 while (current_ptr != NULL) {
476 parent_ptr = current_ptr;
477
478 if (KEY_IS_STRICTLY_LESS(node_ptr->key, current_ptr->key)) {
479 current_ptr = current_ptr->left_ptr;
480 }
481 else {
482 current_ptr = current_ptr->right_ptr;
483 }
484 }
485
486 node_ptr->left_ptr = node_ptr->right_ptr = NULL;
487 RBTREE_NODE_SET_PARENT_PTR(node_ptr, parent_ptr);
488 RBTREE_NODE_SET_COLOR_TO_RED(node_ptr);
489
490 if (parent_ptr == NULL) {
491 rbtree_ptr->root_ptr = node_ptr;
492 }
493 else {
494 const uint8_t dir = KEY_IS_STRICTLY_LESS(parent_ptr->key, node_ptr->key);
495 parent_ptr->child_ptrs[dir] = node_ptr;
496
497 JOIN(internal, JOIN(RBTREE_NAME, insert_fixup))(rbtree_ptr, node_ptr);
498 }
499
500 rbtree_ptr->count++;
501}

505// move the parent of a node (src) to another node (dest).
506static inline void JOIN(internal, JOIN(RBTREE_NAME, node_transplant))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* src_node,
507 RBTREE_NODE_TYPE* dest_node)
508{
509 assert(rbtree_ptr != NULL);
510 assert(src_node != NULL);
511
512 RBTREE_NODE_TYPE* src_node_parent_ptr = RBTREE_NODE_GET_PARENT_PTR(src_node);
513
514 if (src_node_parent_ptr == NULL) {
515 rbtree_ptr->root_ptr = dest_node;
516 }
517 else {
518 src_node_parent_ptr->child_ptrs[RBTREE_CHILD_DIR(src_node)] = dest_node;
519 }
520
521 if (dest_node != NULL) {
522 RBTREE_NODE_SET_PARENT_PTR(dest_node, src_node_parent_ptr);
523 }
524}

526// rebalance tree after delete. see explanation in the sources linked above. for the special case where:
527// (P's child) was not root and was black and had no children
528static inline void JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* P, int dir)
529{
530 assert(P != NULL);
531 assert(dir == 0 || dir == 1);
532
533 RBTREE_NODE_TYPE* N = NULL; // node (temp)
534 RBTREE_NODE_TYPE* S = NULL; // sibling
535 RBTREE_NODE_TYPE* C = NULL; // close nephew
536 RBTREE_NODE_TYPE* D = NULL; // distant nephew
537
538 do {
539 S = P->child_ptrs[1 - dir];
540 D = S->child_ptrs[1 - dir];
541 C = S->child_ptrs[dir];
542
543 if (RBTREE_NODE_IS_RED(S)) {
544 goto Case_3;
545 }
546 if (D != NULL && RBTREE_NODE_IS_RED(D)) {
547 goto Case_6;
548 }
549 if (C != NULL && RBTREE_NODE_IS_RED(C)) {
550 goto Case_5;
551 }
552 if (RBTREE_NODE_IS_RED(P)) {
553 goto Case_4;
554 }
555 RBTREE_NODE_SET_COLOR_TO_RED(S);
556
557 N = P;
558 P = RBTREE_NODE_GET_PARENT_PTR(N);
559 } while (P != NULL && (dir = RBTREE_CHILD_DIR(N), true));
560
561 return;
562Case_3:
563 RBTREE_ROTATE_DIR(rbtree_ptr, P, dir);
564 RBTREE_NODE_SET_COLOR_TO_RED(P);
565 RBTREE_NODE_SET_COLOR_TO_BLACK(S);
566 S = C;
567 D = S->child_ptrs[1 - dir];
568 if (D != NULL && RBTREE_NODE_IS_RED(D)) {
569 goto Case_6;
570 }
571 C = S->child_ptrs[dir];
572 if (C != NULL && RBTREE_NODE_IS_RED(C)) {
573 goto Case_5;
574 }
575Case_4:
576 RBTREE_NODE_SET_COLOR_TO_RED(S);
577 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
578 return;
579Case_5:
580 RBTREE_ROTATE_DIR(rbtree_ptr, S, 1 - dir);
581 RBTREE_NODE_SET_COLOR_TO_RED(S);
582 RBTREE_NODE_SET_COLOR_TO_BLACK(C);
583 D = S;
584 S = C;
585Case_6:
586 RBTREE_ROTATE_DIR(rbtree_ptr, P, dir);
587 RBTREE_NODE_SET_COLOR_TO_COLOR_OF_OTHER(S, P);
588 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
589 RBTREE_NODE_SET_COLOR_TO_BLACK(D);
590 return;
591}

605static inline RBTREE_NODE_TYPE* JOIN(RBTREE_NAME, delete_node)(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* node_ptr)
606{
607 assert(rbtree_ptr != NULL);
608 assert(node_ptr != NULL);
609
610 if (node_ptr->left_ptr == NULL && node_ptr->right_ptr == NULL) {
611 if (node_ptr == rbtree_ptr->root_ptr || RBTREE_NODE_IS_RED(node_ptr)) {
612 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, NULL);
613 }
614 else {
615 RBTREE_NODE_TYPE* const parent_ptr = RBTREE_NODE_GET_PARENT_PTR(node_ptr);
616 const int dir = RBTREE_CHILD_DIR(node_ptr);
617
618 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, NULL);
619
620 JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(rbtree_ptr, parent_ptr, dir);
621 }
622 }
623 else if (node_ptr->left_ptr == NULL || node_ptr->right_ptr == NULL) {
624 const uint8_t dir = node_ptr->left_ptr == NULL;
625
626 assert(RBTREE_NODE_IS_BLACK(node_ptr));
627 assert(RBTREE_NODE_IS_RED(node_ptr->child_ptrs[dir]));
628
629 RBTREE_NODE_SET_COLOR_TO_BLACK(node_ptr->child_ptrs[dir]);
630
631 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, node_ptr->child_ptrs[dir]);
632 }
633 else {
634 RBTREE_NODE_TYPE* successor_ptr = node_ptr->right_ptr;
635
636 while (successor_ptr->left_ptr != NULL) {
637 successor_ptr = successor_ptr->left_ptr;
638 }
639
640 const bool prev_successor_black = RBTREE_NODE_IS_BLACK(successor_ptr);
641 const int prev_successor_dir = RBTREE_CHILD_DIR(successor_ptr);
642 RBTREE_NODE_TYPE* const prev_successor_parent_ptr = RBTREE_NODE_GET_PARENT_PTR(successor_ptr);
643 RBTREE_NODE_TYPE* const prev_successor_child_ptr = successor_ptr->right_ptr;
644
645 {
646 if (RBTREE_NODE_GET_PARENT_PTR(successor_ptr) != node_ptr) {
647 RBTREE_NODE_TRANSPLANT(rbtree_ptr, successor_ptr, successor_ptr->right_ptr);
648
649 successor_ptr->right_ptr = node_ptr->right_ptr;
650 RBTREE_NODE_SET_PARENT_PTR(node_ptr->right_ptr, successor_ptr);
651 }
652 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, successor_ptr);
653
654 successor_ptr->left_ptr = node_ptr->left_ptr;
655 RBTREE_NODE_SET_PARENT_PTR(node_ptr->left_ptr, successor_ptr);
656 }
657
658 {
659 RBTREE_NODE_SET_COLOR_TO_COLOR_OF_OTHER(successor_ptr, node_ptr);
660
661 if (prev_successor_child_ptr != NULL) {
662 assert(prev_successor_black);
663 assert(RBTREE_NODE_IS_RED(prev_successor_child_ptr));
664
665 RBTREE_NODE_SET_COLOR_TO_BLACK(prev_successor_child_ptr);
666 }
667 else if (prev_successor_child_ptr == NULL && prev_successor_black) {
668 if (prev_successor_parent_ptr == node_ptr) {
669 JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(rbtree_ptr, successor_ptr, prev_successor_dir);
670 }
671 else {
672 JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(rbtree_ptr, prev_successor_parent_ptr, prev_successor_dir);
673 }
674 }
675 }
676 }
677
678 node_ptr->left_ptr = node_ptr->right_ptr = NULL;
679 RBTREE_NODE_SET_PARENT_PTR(node_ptr, NULL);
680
681 rbtree_ptr->count--;
682
683 return node_ptr;
684}
