297static inline bool JOIN(RBTREE_NAME, contains_key)(const RBTREE_TYPE* rbtree_ptr, const KEY_TYPE key)
298{
299 assert(rbtree_ptr != NULL);
300
301 RBTREE_NODE_TYPE* node_ptr = rbtree_ptr->root_ptr;
302
303 while (node_ptr != NULL) {
304 const bool is_strictly_less = KEY_IS_STRICTLY_LESS(key, node_ptr->key);
305 const bool is_strictly_greater = KEY_IS_STRICTLY_LESS(node_ptr->key, key);
306
307 if (!is_strictly_less && !is_strictly_greater) {
308 return true;
309 }
310 else if (is_strictly_less) {
311 node_ptr = node_ptr->left_ptr;
312 }
313 else {
314 node_ptr = node_ptr->right_ptr;
315 }
316 }
317 return false;
318}

355// rotate a subtree around a given subtree root node and direction (0: left or 1: right). returns the new subtree root
356static inline RBTREE_NODE_TYPE* JOIN(internal, JOIN(RBTREE_NAME, rotate_dir))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* P,
357 const uint8_t dir)
358{
359 assert(rbtree_ptr != NULL);
360 assert(P != NULL);
361 assert(dir == 0 || dir == 1);
362
363 /* Right rotate around P:
364 P S
365 / \ / \
366 S T --> .. P
367 / \ / \ ⟳ / \
368 .. C .. .. C T
369 / \ / \ / \
370 .. .. .. .. .. ..
371
372 Left rotate around P:
373 P S
374 / \ / \
375 T S P ..
376 / \ / \ --> / \
377 .. .. C .. ↺ T C
378 / \ / \ / \
379 .. .. .. .. .. ..
380 */
381
382 RBTREE_NODE_TYPE* G = RBTREE_NODE_GET_PARENT_PTR(P);
383 RBTREE_NODE_TYPE* S = P->child_ptrs[1 - dir];
384 assert(S != NULL);
385 RBTREE_NODE_TYPE* C = S->child_ptrs[dir];
386
387 P->child_ptrs[1 - dir] = C;
388 if (C != NULL) {
389 RBTREE_NODE_SET_PARENT_PTR(C, P);
390 }
391
392 S->child_ptrs[dir] = P;
393 RBTREE_NODE_SET_PARENT_PTR(P, S);
394
395 RBTREE_NODE_SET_PARENT_PTR(S, G);
396 if (G != NULL) {
397 G->child_ptrs[P == G->left_ptr ? 0 : 1] = S;
398 }
399 else {
400 rbtree_ptr->root_ptr = S;
401 }
402 return S;
403}

405// rebalance tree after insert. see explanation in the sources linked above.
406static inline void JOIN(internal, JOIN(RBTREE_NAME, insert_fixup))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* N)
407{
408 assert(rbtree_ptr != NULL);
409 assert(N != NULL);
410 assert(RBTREE_NODE_IS_RED(N));
411 assert(N != rbtree_ptr->root_ptr);
412
413 RBTREE_NODE_TYPE* P = RBTREE_NODE_GET_PARENT_PTR(N); // parent
414 RBTREE_NODE_TYPE* G = NULL; // grandparent
415 RBTREE_NODE_TYPE* U = NULL; // uncle
416
417 do {
418 if (RBTREE_NODE_IS_BLACK(P)) {
419 return;
420 }
421 if ((G = RBTREE_NODE_GET_PARENT_PTR(P)) == NULL) {
422 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
423 return;
424 }
425 const uint8_t dir = RBTREE_CHILD_DIR(P);
426 U = G->child_ptrs[1 - dir];
427
428 if (U == NULL || RBTREE_NODE_IS_BLACK(U)) {
429 if (N == P->child_ptrs[1 - dir]) {
430 RBTREE_ROTATE_DIR(rbtree_ptr, P, dir);
431 N = P;
432 P = G->child_ptrs[dir];
433 }
434 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
435 RBTREE_NODE_SET_COLOR_TO_RED(G);
436 RBTREE_ROTATE_DIR(rbtree_ptr, G, 1 - dir);
437 return;
438 }
439
440 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
441 RBTREE_NODE_SET_COLOR_TO_BLACK(U);
442 RBTREE_NODE_SET_COLOR_TO_RED(G);
443
444 N = G;
445 P = RBTREE_NODE_GET_PARENT_PTR(N);
446 } while (P != NULL);
447}

461static inline void JOIN(RBTREE_NAME, insert_node)(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* node_ptr)
462{
463 assert(rbtree_ptr != NULL);
464 assert(node_ptr != NULL);
465 assert(RBTREE_CONTAINS_KEY(rbtree_ptr, node_ptr->key) == false);
466
467 RBTREE_NODE_TYPE* parent_ptr = NULL;
468 RBTREE_NODE_TYPE* current_ptr = rbtree_ptr->root_ptr;
469
470 while (current_ptr != NULL) {
471 parent_ptr = current_ptr;
472
473 if (KEY_IS_STRICTLY_LESS(node_ptr->key, current_ptr->key)) {
474 current_ptr = current_ptr->left_ptr;
475 }
476 else {
477 current_ptr = current_ptr->right_ptr;
478 }
479 }
480
481 node_ptr->left_ptr = node_ptr->right_ptr = NULL;
482 RBTREE_NODE_SET_PARENT_PTR(node_ptr, parent_ptr);
483 RBTREE_NODE_SET_COLOR_TO_RED(node_ptr);
484
485 if (parent_ptr == NULL) {
486 rbtree_ptr->root_ptr = node_ptr;
487 }
488 else {
489 const uint8_t dir = KEY_IS_STRICTLY_LESS(parent_ptr->key, node_ptr->key);
490 parent_ptr->child_ptrs[dir] = node_ptr;
491
492 JOIN(internal, JOIN(RBTREE_NAME, insert_fixup))(rbtree_ptr, node_ptr);
493 }
494
495 rbtree_ptr->count++;
496}

500// move the parent of a node (src) to another node (dest).
501static inline void JOIN(internal, JOIN(RBTREE_NAME, node_transplant))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* src_node,
502 RBTREE_NODE_TYPE* dest_node)
503{
504 assert(rbtree_ptr != NULL);
505 assert(src_node != NULL);
506
507 RBTREE_NODE_TYPE* src_node_parent_ptr = RBTREE_NODE_GET_PARENT_PTR(src_node);
508
509 if (src_node_parent_ptr == NULL) {
510 rbtree_ptr->root_ptr = dest_node;
511 }
512 else {
513 src_node_parent_ptr->child_ptrs[RBTREE_CHILD_DIR(src_node)] = dest_node;
514 }
515
516 if (dest_node != NULL) {
517 RBTREE_NODE_SET_PARENT_PTR(dest_node, src_node_parent_ptr);
518 }
519}

521// rebalance tree after delete. see explanation in the sources linked above. for the special case where:
522// (P's child) was not root and was black and had no children
523static inline void JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* P, int dir)
524{
525 assert(P != NULL);
526 assert(dir == 0 || dir == 1);
527
528 RBTREE_NODE_TYPE* N = NULL; // node (temp)
529 RBTREE_NODE_TYPE* S = NULL; // sibling
530 RBTREE_NODE_TYPE* C = NULL; // close nephew
531 RBTREE_NODE_TYPE* D = NULL; // distant nephew
532
533 do {
534 S = P->child_ptrs[1 - dir];
535 D = S->child_ptrs[1 - dir];
536 C = S->child_ptrs[dir];
537
538 if (RBTREE_NODE_IS_RED(S)) {
539 goto Case_3;
540 }
541 if (D != NULL && RBTREE_NODE_IS_RED(D)) {
542 goto Case_6;
543 }
544 if (C != NULL && RBTREE_NODE_IS_RED(C)) {
545 goto Case_5;
546 }
547 if (RBTREE_NODE_IS_RED(P)) {
548 goto Case_4;
549 }
550 RBTREE_NODE_SET_COLOR_TO_RED(S);
551
552 N = P;
553 P = RBTREE_NODE_GET_PARENT_PTR(N);
554 } while (P != NULL && (dir = RBTREE_CHILD_DIR(N), true));
555
556 return;
557Case_3:
558 RBTREE_ROTATE_DIR(rbtree_ptr, P, dir);
559 RBTREE_NODE_SET_COLOR_TO_RED(P);
560 RBTREE_NODE_SET_COLOR_TO_BLACK(S);
561 S = C;
562 D = S->child_ptrs[1 - dir];
563 if (D != NULL && RBTREE_NODE_IS_RED(D)) {
564 goto Case_6;
565 }
566 C = S->child_ptrs[dir];
567 if (C != NULL && RBTREE_NODE_IS_RED(C)) {
568 goto Case_5;
569 }
570Case_4:
571 RBTREE_NODE_SET_COLOR_TO_RED(S);
572 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
573 return;
574Case_5:
575 RBTREE_ROTATE_DIR(rbtree_ptr, S, 1 - dir);
576 RBTREE_NODE_SET_COLOR_TO_RED(S);
577 RBTREE_NODE_SET_COLOR_TO_BLACK(C);
578 D = S;
579 S = C;
580Case_6:
581 RBTREE_ROTATE_DIR(rbtree_ptr, P, dir);
582 RBTREE_NODE_SET_COLOR_TO_COLOR_OF_OTHER(S, P);
583 RBTREE_NODE_SET_COLOR_TO_BLACK(P);
584 RBTREE_NODE_SET_COLOR_TO_BLACK(D);
585 return;
586}

600static inline RBTREE_NODE_TYPE* JOIN(RBTREE_NAME, delete_node)(RBTREE_TYPE* rbtree_ptr, RBTREE_NODE_TYPE* node_ptr)
601{
602 assert(rbtree_ptr != NULL);
603 assert(node_ptr != NULL);
604
605 if (node_ptr->left_ptr == NULL && node_ptr->right_ptr == NULL) {
606 if (node_ptr == rbtree_ptr->root_ptr || RBTREE_NODE_IS_RED(node_ptr)) {
607 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, NULL);
608 }
609 else {
610 RBTREE_NODE_TYPE* const parent_ptr = RBTREE_NODE_GET_PARENT_PTR(node_ptr);
611 const int dir = RBTREE_CHILD_DIR(node_ptr);
612
613 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, NULL);
614
615 JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(rbtree_ptr, parent_ptr, dir);
616 }
617 }
618 else if (node_ptr->left_ptr == NULL || node_ptr->right_ptr == NULL) {
619 const uint8_t dir = node_ptr->left_ptr == NULL;
620
621 assert(RBTREE_NODE_IS_BLACK(node_ptr));
622 assert(RBTREE_NODE_IS_RED(node_ptr->child_ptrs[dir]));
623
624 RBTREE_NODE_SET_COLOR_TO_BLACK(node_ptr->child_ptrs[dir]);
625
626 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, node_ptr->child_ptrs[dir]);
627 }
628 else {
629 RBTREE_NODE_TYPE* successor_ptr = node_ptr->right_ptr;
630
631 while (successor_ptr->left_ptr != NULL) {
632 successor_ptr = successor_ptr->left_ptr;
633 }
634
635 const bool prev_successor_black = RBTREE_NODE_IS_BLACK(successor_ptr);
636 const int prev_successor_dir = RBTREE_CHILD_DIR(successor_ptr);
637 RBTREE_NODE_TYPE* const prev_successor_parent_ptr = RBTREE_NODE_GET_PARENT_PTR(successor_ptr);
638 RBTREE_NODE_TYPE* const prev_successor_child_ptr = successor_ptr->right_ptr;
639
640 {
641 if (RBTREE_NODE_GET_PARENT_PTR(successor_ptr) != node_ptr) {
642 RBTREE_NODE_TRANSPLANT(rbtree_ptr, successor_ptr, successor_ptr->right_ptr);
643
644 successor_ptr->right_ptr = node_ptr->right_ptr;
645 RBTREE_NODE_SET_PARENT_PTR(node_ptr->right_ptr, successor_ptr);
646 }
647 RBTREE_NODE_TRANSPLANT(rbtree_ptr, node_ptr, successor_ptr);
648
649 successor_ptr->left_ptr = node_ptr->left_ptr;
650 RBTREE_NODE_SET_PARENT_PTR(node_ptr->left_ptr, successor_ptr);
651 }
652
653 {
654 RBTREE_NODE_SET_COLOR_TO_COLOR_OF_OTHER(successor_ptr, node_ptr);
655
656 if (prev_successor_child_ptr != NULL) {
657 assert(prev_successor_black);
658 assert(RBTREE_NODE_IS_RED(prev_successor_child_ptr));
659
660 RBTREE_NODE_SET_COLOR_TO_BLACK(prev_successor_child_ptr);
661 }
662 else if (prev_successor_child_ptr == NULL && prev_successor_black) {
663 if (prev_successor_parent_ptr == node_ptr) {
664 JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(rbtree_ptr, successor_ptr, prev_successor_dir);
665 }
666 else {
667 JOIN(internal, JOIN(RBTREE_NAME, delete_fixup))(rbtree_ptr, prev_successor_parent_ptr, prev_successor_dir);
668 }
669 }
670 }
671 }
672
673 node_ptr->left_ptr = node_ptr->right_ptr = NULL;
674 RBTREE_NODE_SET_PARENT_PTR(node_ptr, NULL);
675
676 rbtree_ptr->count--;
677
678 return node_ptr;
679}
