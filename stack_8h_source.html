276static inline bool JOIN(__STACK_PREFIX, push)(STACK_TYPE* stack_ptr, VALUE_TYPE value) {
277 assert(NULL != stack_ptr);
278
279 if (stack_ptr->count + 1 == stack_ptr->capacity) {
280 if (stack_ptr->capacity > SIZE_MAX / sizeof(VALUE_TYPE) / 2) {
281 return false;
282 }
283 void* temp_ptr = stack_ptr->allocator_ops.reallocate_f_p(stack_ptr->allocator_context_ptr, stack_ptr->values,
284 alignof(VALUE_TYPE), sizeof(VALUE_TYPE) * stack_ptr->capacity,
285 2 * sizeof(VALUE_TYPE) * stack_ptr->capacity);
286 if (temp_ptr == NULL) {
287 return false;
288 }
289 stack_ptr->values = (VALUE_TYPE*)temp_ptr;
290 stack_ptr->capacity *= 2;
291 }
292 stack_ptr->values[stack_ptr->count++] = value;
293
294 return true;
295}
