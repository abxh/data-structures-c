282static inline bool JOIN(STACK_PREFIX, grow)(STACK_TYPE* stack_ptr, size_t new_capacity) {
283 assert(NULL != stack_ptr);
284
285 if (new_capacity <= stack_ptr->capacity || new_capacity > SIZE_MAX / sizeof(VALUE_TYPE)) {
286 return false;
287 }
288 void* temp_ptr =

290 sizeof(VALUE_TYPE) * stack_ptr->capacity, sizeof(VALUE_TYPE) * new_capacity);
291 if (temp_ptr == NULL) {
