262static inline bool JOIN(__STACK_PREFIX, push)(STACK_TYPE* stack_ptr, VALUE_TYPE value) {
263 assert(NULL != stack_ptr);
264
265 if (stack_ptr->count == stack_ptr->capacity) {
266 if (stack_ptr->capacity > SIZE_MAX / sizeof(VALUE_TYPE) / 2) {
267 return false;
268 }
269 void* temp_ptr = stack_ptr->allocator_ops.reallocate_f_p(stack_ptr->allocator_context_ptr, stack_ptr->values,
270 alignof(VALUE_TYPE), sizeof(VALUE_TYPE) * stack_ptr->capacity,
271 2 * sizeof(VALUE_TYPE) * stack_ptr->capacity);
272 if (temp_ptr == NULL) {
273 return false;
274 }
275 stack_ptr->values = (VALUE_TYPE*)temp_ptr;
276 stack_ptr->capacity *= 2;
277 }
278 stack_ptr->values[stack_ptr->count++] = value;
279
280 return true;
281}
