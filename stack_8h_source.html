280static inline bool JOIN(__STACK_PREFIX, push)(STACK_TYPE* stack_ptr, VALUE_TYPE value) {
281 assert(NULL != stack_ptr);
282
283 if (stack_ptr->count == stack_ptr->capacity) {
284 if (stack_ptr->capacity > SIZE_MAX / sizeof(VALUE_TYPE) / 2) {
285 return false;
286 }
287 void* temp_ptr = stack_ptr->allocator_ops.reallocate_f_p(stack_ptr->allocator_context_ptr, stack_ptr->values,
288 alignof(VALUE_TYPE), sizeof(VALUE_TYPE) * stack_ptr->capacity,
289 2 * sizeof(VALUE_TYPE) * stack_ptr->capacity);
290 if (temp_ptr == NULL) {
291 return false;
292 }
293 stack_ptr->values = (VALUE_TYPE*)temp_ptr;
294 stack_ptr->capacity *= 2;
295 }
296 stack_ptr->values[stack_ptr->count++] = value;
297
298 return true;
299}
