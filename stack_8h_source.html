118 if (!stack_ptr) {

284static inline bool JOIN(STACK_PREFIX, grow)(STACK_TYPE* stack_ptr, size_t new_capacity) {
285 assert(NULL != stack_ptr);
286
287 if (new_capacity <= stack_ptr->capacity || new_capacity > SIZE_MAX / sizeof(VALUE_TYPE)) {
288 return false;
289 }
290 void* temp_ptr =

292 sizeof(VALUE_TYPE) * stack_ptr->capacity, sizeof(VALUE_TYPE) * new_capacity);
293 if (temp_ptr == NULL) {

315static inline bool JOIN(STACK_PREFIX, push)(STACK_TYPE* stack_ptr, VALUE_TYPE value) {
