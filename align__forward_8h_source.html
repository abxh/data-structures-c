36static inline uintptr_t align_forward(const uintptr_t ptr, const size_t align)
37{
38 assert(is_pow2(align));
39
40 const uintptr_t p = ptr;
41 const uintptr_t a = (uintptr_t)align;
42 const uintptr_t r = p & (a - 1); // Same as (p % a) but faster as 'a' is a power of two
43
44 // % is defined so for r = (p % a), q: some integer
45 // p = q * a + r
46
47 // assuming r != 0:
48 //
49 // (q+1)*a
50 // ----------->
51 // r a-r
52 // ----> ----->
53 // q*a p p+a-r
54 // <------|----|------|------>
55
56 if (r != 0) {
57 // If 'p' address is not aligned, push the address to the
58 // next value which is aligned
59 return p + a - r;
60 }
61 else {
62 return p;
63 }
64}
