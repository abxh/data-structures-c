44static inline uintptr_t align_forward(const uintptr_t ptr, const size_t align) {
45 assert(is_pow2(align));
46
47 const uintptr_t p = ptr;
48 const uintptr_t a = (uintptr_t)align;
49 const uintptr_t r = p & (a - 1); // Same as (p % a) but faster as 'a' is a power of two
50
51 // % is defined so for r = (p % a), q: some integer
52 // p = q * a + r
53 // <=> r = p - q * a
54
55 // assuming r != 0:
56 //
57 // q*(a+1)
58 // ----------->
59 // r a-r
60 // ----> ----->
61 // q*a p p+a-r
62 // <------|----|------|------>
63
64 if (r != 0) {
65 // If 'p' address is not aligned, push the address to the
66 // next value which is aligned
67 return p + a - r;
68 } else {
69 return p;
70 }
71}
